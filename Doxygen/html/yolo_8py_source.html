72class Detect(nn.Module):
73 # YOLOv5 Detect head for detection models
74 stride = None # strides computed during build
75 dynamic = False # force grid reconstruction
76 export = False # export mode
77
78 def __init__(self, nc=80, anchors=(), ch=(), inplace=True): # detection layer
79 super().__init__()
80 self.nc = nc # number of classes
81 self.no = nc + 5 # number of outputs per anchor
82 self.nl = len(anchors) # number of detection layers
83 self.na = len(anchors[0]) // 2 # number of anchors
84 self.grid = [torch.empty(0) for _ in range(self.nl)] # init grid
85 self.anchor_grid = [torch.empty(0) for _ in range(self.nl)] # init anchor grid
86 self.register_buffer("anchors", torch.tensor(anchors).float().view(self.nl, -1, 2)) # shape(nl,na,2)
87 self.m = nn.ModuleList(nn.Conv2d(x, self.no * self.na, 1) for x in ch) # output conv
88 self.inplace = inplace # use inplace ops (e.g. slice assignment)
89
90 def forward(self, x):
91 z = [] # inference output
92 for i in range(self.nl):
93 x[i] = self.m[i](x[i]) # conv
94 bs, _, ny, nx = x[i].shape # x(bs,255,20,20) to x(bs,3,20,20,85)
95 x[i] = x[i].view(bs, self.na, self.no, ny, nx).permute(0, 1, 3, 4, 2).contiguous()
96
97 if not self.training: # inference
98 if self.dynamic or self.grid[i].shape[2:4] != x[i].shape[2:4]:
99 self.grid[i], self.anchor_grid[i] = self._make_grid(nx, ny, i)
100
101 if isinstance(self, Segment): # (boxes + masks)
102 xy, wh, conf, mask = x[i].split((2, 2, self.nc + 1, self.no - self.nc - 5), 4)
103 xy = (xy.sigmoid() * 2 + self.grid[i]) * self.stride[i] # xy
104 wh = (wh.sigmoid() * 2) ** 2 * self.anchor_grid[i] # wh
105 y = torch.cat((xy, wh, conf.sigmoid(), mask), 4)
106 else: # Detect (boxes only)
107 xy, wh, conf = x[i].sigmoid().split((2, 2, self.nc + 1), 4)
108 xy = (xy * 2 + self.grid[i]) * self.stride[i] # xy
109 wh = (wh * 2) ** 2 * self.anchor_grid[i] # wh
110 y = torch.cat((xy, wh, conf), 4)
111 z.append(y.view(bs, self.na * nx * ny, self.no))
112
113 return x if self.training else (torch.cat(z, 1),) if self.export else (torch.cat(z, 1), x)
114
115 def _make_grid(self, nx=20, ny=20, i=0, torch_1_10=check_version(torch.__version__, "1.10.0")):
116 d = self.anchors[i].device
117 t = self.anchors[i].dtype
118 shape = 1, self.na, ny, nx, 2 # grid shape
119 y, x = torch.arange(ny, device=d, dtype=t), torch.arange(nx, device=d, dtype=t)
120 yv, xv = torch.meshgrid(y, x, indexing="ij") if torch_1_10 else torch.meshgrid(y, x) # torch>=0.7 compatibility
121 grid = torch.stack((xv, yv), 2).expand(shape) - 0.5 # add grid offset, i.e. y = 2.0 * x - 0.5
122 anchor_grid = (self.anchors[i] * self.stride[i]).view((1, self.na, 1, 1, 2)).expand(shape)
123 return grid, anchor_grid

126class Segment(Detect):
127 # YOLOv5 Segment head for segmentation models
128 def __init__(self, nc=80, anchors=(), nm=32, npr=256, ch=(), inplace=True):
129 super().__init__(nc, anchors, ch, inplace)
130 self.nm = nm # number of masks
131 self.npr = npr # number of protos
132 self.nono = 5 + nc + self.nm # number of outputs per anchor
133 self.mm = nn.ModuleList(nn.Conv2d(x, self.nono * self.na, 1) for x in ch) # output conv
134 self.proto = Proto(ch[0], self.npr, self.nm) # protos
135 self.detect = Detect.forward
136
137 def forward(self, x):
138 p = self.proto(x[0])
139 x = self.detect(self, x)
140 return (x, p) if self.training else (x[0], p) if self.export else (x[0], p, x[1])

143class BaseModel(nn.Module):
144 # YOLOv5 base model
145 def forward(self, x, profile=False, visualize=False):
146 return self._forward_once(x, profile, visualize) # single-scale inference, train
147
148 def _forward_once(self, x, profile=False, visualize=False):
149 y, dt = [], [] # outputs
150 for m in self.model:
151 if m.f != -1: # if not from previous layer
152 x = y[m.f] if isinstance(m.f, int) else [x if j == -1 else y[j] for j in m.f] # from earlier layers
153 if profile:
154 self._profile_one_layer(m, x, dt)
155 x = m(x) # run
156 y.append(x if m.i in self.save else None) # save output
157 if visualize:
158 feature_visualization(x, m.type, m.i, save_dir=visualize)
159 return x
160
161 def _profile_one_layer(self, m, x, dt):
162 c = m == self.model[-1] # is final layer, copy input as inplace fix
163 o = thop.profile(m, inputs=(x.copy() if c else x,), verbose=False)[0] / 1e9 * 2 if thop else 0 # FLOPs
164 t = time_sync()
165 for _ in range(10):
166 m(x.copy() if c else x)
167 dt.append((time_sync() - t) * 100)
168 if m == self.model[0]:
169 LOGGER.info(f"{'time (ms)':>10s} {'GFLOPs':>10s} {'params':>10s} module")
170 LOGGER.info(f"{dt[-1]:10.2f} {o:10.2f} {m.np:10.0f} {m.type}")
171 if c:
172 LOGGER.info(f"{sum(dt):10.2f} {'-':>10s} {'-':>10s} Total")
173
174 def fuse(self): # fuse model Conv2d() + BatchNorm2d() layers
175 LOGGER.info("Fusing layers... ")
176 for m in self.model.modules():
177 if isinstance(m, (Conv, DWConv)) and hasattr(m, "bn"):
178 m.conv = fuse_conv_and_bn(m.conv, m.bn) # update conv
179 delattr(m, "bn") # remove batchnorm
180 m.forward = m.forward_fuse # update forward
181 self.info()
182 return self
183
184 def info(self, verbose=False, img_size=640): # print model information
185 model_info(self, verbose, img_size)
186
187 def _apply(self, fn):
188 # Apply to(), cpu(), cuda(), half() to model tensors that are not parameters or registered buffers
189 self = super()._apply(fn)
190 m = self.model[-1] # Detect()
191 if isinstance(m, (Detect, Segment)):
192 m.stride = fn(m.stride)
193 m.grid = list(map(fn, m.grid))
194 if isinstance(m.anchor_grid, list):
195 m.anchor_grid = list(map(fn, m.anchor_grid))
196 return self

199class DetectionModel(BaseModel):
200 # YOLOv5 detection model
201 def __init__(self, cfg="yolov5s.yaml", ch=3, nc=None, anchors=None): # model, input channels, number of classes
202 super().__init__()
203 if isinstance(cfg, dict):
204 self.yaml = cfg # model dict
205 else: # is *.yaml
206 import yaml # for torch hub
207
208 self.yaml_file = Path(cfg).name
209 with open(cfg, encoding="ascii", errors="ignore") as f:
210 self.yaml = yaml.safe_load(f) # model dict
211
212 # Define model
213 ch = self.yaml["ch"] = self.yaml.get("ch", ch) # input channels
214 if nc and nc != self.yaml["nc"]:
215 LOGGER.info(f"Overriding model.yaml nc={self.yaml['nc']} with nc={nc}")
216 self.yaml["nc"] = nc # override yaml value
217 if anchors:
218 LOGGER.info(f"Overriding model.yaml anchors with anchors={anchors}")
219 self.yaml["anchors"] = round(anchors) # override yaml value
220 self.model, self.save = parse_model(deepcopy(self.yaml), ch=[ch]) # model, savelist
221 self.names = [str(i) for i in range(self.yaml["nc"])] # default names
222 self.inplace = self.yaml.get("inplace", True)
223
224 # Build strides, anchors
225 m = self.model[-1] # Detect()
226 if isinstance(m, (Detect, Segment)):
227 s = 256 # 2x min stride
228 m.inplace = self.inplace
229 forward = lambda x: self.forwardforward(x)[0] if isinstance(m, Segment) else self.forwardforward(x)
230 m.stride = torch.tensor([s / x.shape[-2] for x in forward(torch.zeros(1, ch, s, s))]) # forward
231 check_anchor_order(m)
232 m.anchors /= m.stride.view(-1, 1, 1)
233 self.stride = m.stride
234 self._initialize_biases() # only run once
235
236 # Init weights, biases
237 initialize_weights(self)
238 self.info()
239 LOGGER.info("")

241 def forward(self, x, augment=False, profile=False, visualize=False):
242 if augment:
243 return self._forward_augment(x) # augmented inference, None
244 return self._forward_once(x, profile, visualize) # single-scale inference, train
245
246 def _forward_augment(self, x):
247 img_size = x.shape[-2:] # height, width
248 s = [1, 0.83, 0.67] # scales
249 f = [None, 3, None] # flips (2-ud, 3-lr)
250 y = [] # outputs
251 for si, fi in zip(s, f):
252 xi = scale_img(x.flip(fi) if fi else x, si, gs=int(self.stride.max()))
253 yi = self._forward_once(xi)[0] # forward
254 # cv2.imwrite(f'img_{si}.jpg', 255 * xi[0].cpu().numpy().transpose((1, 2, 0))[:, :, ::-1]) # save
255 yi = self._descale_pred(yi, fi, si, img_size)
256 y.append(yi)
257 y = self._clip_augmented(y) # clip augmented tails
258 return torch.cat(y, 1), None # augmented inference, train
259
260 def _descale_pred(self, p, flips, scale, img_size):
261 # de-scale predictions following augmented inference (inverse operation)
262 if self.inplace:
263 p[..., :4] /= scale # de-scale
264 if flips == 2:
265 p[..., 1] = img_size[0] - p[..., 1] # de-flip ud
266 elif flips == 3:
267 p[..., 0] = img_size[1] - p[..., 0] # de-flip lr
268 else:
269 x, y, wh = p[..., 0:1] / scale, p[..., 1:2] / scale, p[..., 2:4] / scale # de-scale
270 if flips == 2:
271 y = img_size[0] - y # de-flip ud
272 elif flips == 3:
273 x = img_size[1] - x # de-flip lr
274 p = torch.cat((x, y, wh, p[..., 4:]), -1)
275 return p
276
277 def _clip_augmented(self, y):
278 # Clip YOLOv5 augmented inference tails
279 nl = self.model[-1].nl # number of detection layers (P3-P5)
280 g = sum(4**x for x in range(nl)) # grid points
281 e = 1 # exclude layer count
282 i = (y[0].shape[1] // g) * sum(4**x for x in range(e)) # indices
283 y[0] = y[0][:, :-i] # large
284 i = (y[-1].shape[1] // g) * sum(4 ** (nl - 1 - x) for x in range(e)) # indices
285 y[-1] = y[-1][:, i:] # small
286 return y
287
288 def _initialize_biases(self, cf=None): # initialize biases into Detect(), cf is class frequency
289 # https://arxiv.org/abs/1708.02002 section 3.3
290 # cf = torch.bincount(torch.tensor(np.concatenate(dataset.labels, 0)[:, 0]).long(), minlength=nc) + 1.
291 m = self.model[-1] # Detect() module
292 for mi, s in zip(m.m, m.stride): # from
293 b = mi.bias.view(m.na, -1) # conv.bias(255) to (3,85)
294 b.data[:, 4] += math.log(8 / (640 / s) ** 2) # obj (8 objects per 640 image)
295 b.data[:, 5 : 5 + m.nc] += (
296 math.log(0.6 / (m.nc - 0.99999)) if cf is None else torch.log(cf / cf.sum())
297 ) # cls
298 mi.bias = torch.nn.Parameter(b.view(-1), requires_grad=True)

336def parse_model(d, ch): # model_dict, input_channels(3)
337 # Parse a YOLOv5 model.yaml dictionary
338 LOGGER.info(f"\n{'':>3}{'from':>18}{'n':>3}{'params':>10} {'module':<40}{'arguments':<30}")
339 anchors, nc, gd, gw, act, ch_mul = (
340 d["anchors"],
341 d["nc"],
342 d["depth_multiple"],
343 d["width_multiple"],
344 d.get("activation"),
345 d.get("channel_multiple"),
346 )
347 if act:
348 Conv.default_act = eval(act) # redefine default activation, i.e. Conv.default_act = nn.SiLU()
349 LOGGER.info(f"{colorstr('activation:')} {act}") # print
350 if not ch_mul:
351 ch_mul = 8
352 na = (len(anchors[0]) // 2) if isinstance(anchors, list) else anchors # number of anchors
353 no = na * (nc + 5) # number of outputs = anchors * (classes + 5)
354
355 layers, save, c2 = [], [], ch[-1] # layers, savelist, ch out
356 for i, (f, n, m, args) in enumerate(d["backbone"] + d["head"]): # from, number, module, args
357 m = eval(m) if isinstance(m, str) else m # eval strings
358 for j, a in enumerate(args):
359 with contextlib.suppress(NameError):
360 args[j] = eval(a) if isinstance(a, str) else a # eval strings
361
362 n = n_ = max(round(n * gd), 1) if n > 1 else n # depth gain
363 if m in {
364 Conv,
365 GhostConv,
366 Bottleneck,
367 GhostBottleneck,
368 SPP,
369 SPPF,
370 DWConv,
371 MixConv2d,
372 Focus,
373 CrossConv,
374 BottleneckCSP,
375 C3,
376 C3TR,
377 C3SPP,
378 C3Ghost,
379 nn.ConvTranspose2d,
380 DWConvTranspose2d,
381 C3x,
382 }:
383 c1, c2 = ch[f], args[0]
384 if c2 != no: # if not output
385 c2 = make_divisible(c2 * gw, ch_mul)
386
387 args = [c1, c2, *args[1:]]
388 if m in {BottleneckCSP, C3, C3TR, C3Ghost, C3x}:
389 args.insert(2, n) # number of repeats
390 n = 1
391 elif m is nn.BatchNorm2d:
392 args = [ch[f]]
393 elif m is Concat:
394 c2 = sum(ch[x] for x in f)
395 # TODO: channel, gw, gd
396 elif m in {Detect, Segment}:
397 args.append([ch[x] for x in f])
398 if isinstance(args[1], int): # number of anchors
399 args[1] = [list(range(args[1] * 2))] * len(f)
400 if m is Segment:
401 args[3] = make_divisible(args[3] * gw, ch_mul)
402 elif m is Contract:
403 c2 = ch[f] * args[0] ** 2
404 elif m is Expand:
405 c2 = ch[f] // args[0] ** 2
406 else:
407 c2 = ch[f]
408
409 m_ = nn.Sequential(*(m(*args) for _ in range(n))) if n > 1 else m(*args) # module
410 t = str(m)[8:-2].replace("__main__.", "") # module type
411 np = sum(x.numel() for x in m_.parameters()) # number params
412 m_.i, m_.f, m_.type, m_.np = i, f, t, np # attach index, 'from' index, type, number params
413 LOGGER.info(f"{i:>3}{str(f):>18}{n_:>3}{np:10.0f} {t:<40}{str(args):<30}") # print
414 save.extend(x % i for x in ([f] if isinstance(f, int) else f) if x != -1) # append to savelist
415 layers.append(m_)
416 if i == 0:
417 ch = []
418 ch.append(c2)
419 return nn.Sequential(*layers), sorted(save)
