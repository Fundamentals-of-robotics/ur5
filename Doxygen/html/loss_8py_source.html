16class BCEBlurWithLogitsLoss(nn.Module):
17 # BCEwithLogitLoss() with reduced missing label effects.
18 def __init__(self, alpha=0.05):
19 super().__init__()
20 self.loss_fcn = nn.BCEWithLogitsLoss(reduction="none") # must be nn.BCEWithLogitsLoss()
21 self.alpha = alpha
22
23 def forward(self, pred, true):
24 loss = self.loss_fcn(pred, true)
25 pred = torch.sigmoid(pred) # prob from logits
26 dx = pred - true # reduce only missing label effects
27 # dx = (pred - true).abs() # reduce missing label and false label effects
28 alpha_factor = 1 - torch.exp((dx - 1) / (self.alpha + 1e-4))
29 loss *= alpha_factor
30 return loss.mean()

33class FocalLoss(nn.Module):
34 # Wraps focal loss around existing loss_fcn(), i.e. criteria = FocalLoss(nn.BCEWithLogitsLoss(), gamma=1.5)
35 def __init__(self, loss_fcn, gamma=1.5, alpha=0.25):
36 super().__init__()
37 self.loss_fcn = loss_fcn # must be nn.BCEWithLogitsLoss()
38 self.gamma = gamma
39 self.alpha = alpha
40 self.reduction = loss_fcn.reduction
41 self.loss_fcn.reduction = "none" # required to apply FL to each element
42
43 def forward(self, pred, true):
44 loss = self.loss_fcn(pred, true)
45 # p_t = torch.exp(-loss)
46 # loss *= self.alpha * (1.000001 - p_t) ** self.gamma # non-zero power for gradient stability
47
48 # TF implementation https://github.com/tensorflow/addons/blob/v0.7.1/tensorflow_addons/losses/focal_loss.py
49 pred_prob = torch.sigmoid(pred) # prob from logits
50 p_t = true * pred_prob + (1 - true) * (1 - pred_prob)
51 alpha_factor = true * self.alpha + (1 - true) * (1 - self.alpha)
52 modulating_factor = (1.0 - p_t) ** self.gamma
53 loss *= alpha_factor * modulating_factor
54
55 if self.reduction == "mean":
56 return loss.mean()
57 elif self.reduction == "sum":
58 return loss.sum()
59 else: # 'none'
60 return loss

63class QFocalLoss(nn.Module):
64 # Wraps Quality focal loss around existing loss_fcn(), i.e. criteria = FocalLoss(nn.BCEWithLogitsLoss(), gamma=1.5)
65 def __init__(self, loss_fcn, gamma=1.5, alpha=0.25):
66 super().__init__()
67 self.loss_fcn = loss_fcn # must be nn.BCEWithLogitsLoss()
68 self.gamma = gamma
69 self.alpha = alpha
70 self.reduction = loss_fcn.reduction
71 self.loss_fcn.reduction = "none" # required to apply FL to each element
72
73 def forward(self, pred, true):
74 loss = self.loss_fcn(pred, true)
75
76 pred_prob = torch.sigmoid(pred) # prob from logits
77 alpha_factor = true * self.alpha + (1 - true) * (1 - self.alpha)
78 modulating_factor = torch.abs(true - pred_prob) ** self.gamma
79 loss *= alpha_factor * modulating_factor
80
81 if self.reduction == "mean":
82 return loss.mean()
83 elif self.reduction == "sum":
84 return loss.sum()
85 else: # 'none'

92 # Compute losses
93 def __init__(self, model, autobalance=False):
94 device = next(model.parameters()).device # get model device
95 h = model.hyp # hyperparameters
96
97 # Define criteria
98 BCEcls = nn.BCEWithLogitsLoss(pos_weight=torch.tensor([h["cls_pw"]], device=device))
99 BCEobj = nn.BCEWithLogitsLoss(pos_weight=torch.tensor([h["obj_pw"]], device=device))
100
101 # Class label smoothing https://arxiv.org/pdf/1902.04103.pdf eqn 3
102 self.cp, self.cn = smooth_BCE(eps=h.get("label_smoothing", 0.0)) # positive, negative BCE targets
103
104 # Focal loss
105 g = h["fl_gamma"] # focal loss gamma
106 if g > 0:
107 BCEcls, BCEobj = FocalLoss(BCEcls, g), FocalLoss(BCEobj, g)
108
109 m = de_parallel(model).model[-1] # Detect() module
110 self.balance = {3: [4.0, 1.0, 0.4]}.get(m.nl, [4.0, 1.0, 0.25, 0.06, 0.02]) # P3-P7
111 self.ssi = list(m.stride).index(16) if autobalance else 0 # stride 16 index
112 self.BCEcls, self.BCEobj, self.gr, self.hyp, self.autobalance = BCEcls, BCEobj, 1.0, h, autobalance
113 self.na = m.na # number of anchors
114 self.nc = m.nc # number of classes
115 self.nl = m.nl # number of layers
116 self.anchors = m.anchors
117 self.device = device

119 def __call__(self, p, targets): # predictions, targets
120 lcls = torch.zeros(1, device=self.device) # class loss
121 lbox = torch.zeros(1, device=self.device) # box loss
122 lobj = torch.zeros(1, device=self.device) # object loss
123 tcls, tbox, indices, anchors = self.build_targets(p, targets) # targets
124
125 # Losses
126 for i, pi in enumerate(p): # layer index, layer predictions
127 b, a, gj, gi = indices[i] # image, anchor, gridy, gridx
128 tobj = torch.zeros(pi.shape[:4], dtype=pi.dtype, device=self.device) # target obj
129
130 n = b.shape[0] # number of targets
131 if n:
132 # pxy, pwh, _, pcls = pi[b, a, gj, gi].tensor_split((2, 4, 5), dim=1) # faster, requires torch 1.8.0
133 pxy, pwh, _, pcls = pi[b, a, gj, gi].split((2, 2, 1, self.nc), 1) # target-subset of predictions
134
135 # Regression
136 pxy = pxy.sigmoid() * 2 - 0.5
137 pwh = (pwh.sigmoid() * 2) ** 2 * anchors[i]
138 pbox = torch.cat((pxy, pwh), 1) # predicted box
139 iou = bbox_iou(pbox, tbox[i], CIoU=True).squeeze() # iou(prediction, target)
140 lbox += (1.0 - iou).mean() # iou loss
141
142 # Objectness
143 iou = iou.detach().clamp(0).type(tobj.dtype)
144 if self.sort_obj_iou:
145 j = iou.argsort()
146 b, a, gj, gi, iou = b[j], a[j], gj[j], gi[j], iou[j]
147 if self.gr < 1:
148 iou = (1.0 - self.gr) + self.gr * iou
149 tobj[b, a, gj, gi] = iou # iou ratio
150
151 # Classification
152 if self.nc > 1: # cls loss (only if multiple classes)
153 t = torch.full_like(pcls, self.cn, device=self.device) # targets
154 t[range(n), tcls[i]] = self.cp
155 lcls += self.BCEcls(pcls, t) # BCE
156
157 # Append targets to text file
158 # with open('targets.txt', 'a') as file:
159 # [file.write('%11.5g ' * 4 % tuple(x) + '\n') for x in torch.cat((txy[i], twh[i]), 1)]
160
161 obji = self.BCEobj(pi[..., 4], tobj)
162 lobj += obji * self.balance[i] # obj loss
163 if self.autobalance:
164 self.balance[i] = self.balance[i] * 0.9999 + 0.0001 / obji.detach().item()
165
166 if self.autobalance:
167 self.balance = [x / self.balance[self.ssi] for x in self.balance]
168 lbox *= self.hyp["box"]
169 lobj *= self.hyp["obj"]
170 lcls *= self.hyp["cls"]
171 bs = tobj.shape[0] # batch size
172
173 return (lbox + lobj + lcls) * bs, torch.cat((lbox, lobj, lcls)).detach()

175 def build_targets(self, p, targets):
176 # Build targets for compute_loss(), input targets(image,class,x,y,w,h)
177 na, nt = self.na, targets.shape[0] # number of anchors, targets
178 tcls, tbox, indices, anch = [], [], [], []
179 gain = torch.ones(7, device=self.device) # normalized to gridspace gain
180 ai = torch.arange(na, device=self.device).float().view(na, 1).repeat(1, nt) # same as .repeat_interleave(nt)
181 targets = torch.cat((targets.repeat(na, 1, 1), ai[..., None]), 2) # append anchor indices
182
183 g = 0.5 # bias
184 off = (
185 torch.tensor(
186 [
187 [0, 0],
188 [1, 0],
189 [0, 1],
190 [-1, 0],
191 [0, -1], # j,k,l,m
192 # [1, 1], [1, -1], [-1, 1], [-1, -1], # jk,jm,lk,lm
193 ],
194 device=self.device,
195 ).float()
196 * g
197 ) # offsets
198
199 for i in range(self.nl):
200 anchors, shape = self.anchors[i], p[i].shape
201 gain[2:6] = torch.tensor(shape)[[3, 2, 3, 2]] # xyxy gain
202
203 # Match targets to anchors
204 t = targets * gain # shape(3,n,7)
205 if nt:
206 # Matches
207 r = t[..., 4:6] / anchors[:, None] # wh ratio
208 j = torch.max(r, 1 / r).max(2)[0] < self.hyp["anchor_t"] # compare
209 # j = wh_iou(anchors, t[:, 4:6]) > model.hyp['iou_t'] # iou(3,n)=wh_iou(anchors(3,2), gwh(n,2))
210 t = t[j] # filter
211
212 # Offsets
213 gxy = t[:, 2:4] # grid xy
214 gxi = gain[[2, 3]] - gxy # inverse
215 j, k = ((gxy % 1 < g) & (gxy > 1)).T
216 l, m = ((gxi % 1 < g) & (gxi > 1)).T
217 j = torch.stack((torch.ones_like(j), j, k, l, m))
218 t = t.repeat((5, 1, 1))[j]
219 offsets = (torch.zeros_like(gxy)[None] + off[:, None])[j]
220 else:
221 t = targets[0]
222 offsets = 0
223
224 # Define
225 bc, gxy, gwh, a = t.chunk(4, 1) # (image, class), grid xy, grid wh, anchors
226 a, (b, c) = a.long().view(-1), bc.long().T # anchors, image, class
227 gij = (gxy - offsets).long()
228 gi, gj = gij.T # grid indices
229
230 # Append
231 indices.append((b, a, gj.clamp_(0, shape[2] - 1), gi.clamp_(0, shape[3] - 1))) # image, anchor, grid
232 tbox.append(torch.cat((gxy - gij, gwh), 1)) # box
233 anch.append(anchors[a]) # anchors
234 tcls.append(c) # class
235
236 return tcls, tbox, indices, anch
