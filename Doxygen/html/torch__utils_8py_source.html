108def select_device(device="", batch_size=0, newline=True):
109 # device = None or 'cpu' or 0 or '0' or '0,1,2,3'
110 s = f"YOLOv5 üöÄ {git_describe() or file_date()} Python-{platform.python_version()} torch-{torch.__version__} "
111 device = str(device).strip().lower().replace("cuda:", "").replace("none", "") # to string, 'cuda:0' to '0'
112 cpu = device == "cpu"
113 mps = device == "mps" # Apple Metal Performance Shaders (MPS)
114 if cpu or mps:
115 os.environ["CUDA_VISIBLE_DEVICES"] = "-1" # force torch.cuda.is_available() = False
116 elif device: # non-cpu device requested
117 os.environ["CUDA_VISIBLE_DEVICES"] = device # set environment variable - must be before assert is_available()
118 assert torch.cuda.is_available() and torch.cuda.device_count() >= len(
119 device.replace(",", "")
120 ), f"Invalid CUDA '--device {device}' requested, use '--device cpu' or pass valid CUDA device(s)"
121
122 if not cpu and not mps and torch.cuda.is_available(): # prefer GPU if available
123 devices = device.split(",") if device else "0" # range(torch.cuda.device_count()) # i.e. 0,1,6,7
124 n = len(devices) # device count
125 if n > 1 and batch_size > 0: # check batch_size is divisible by device_count
126 assert batch_size % n == 0, f"batch-size {batch_size} not multiple of GPU count {n}"
127 space = " " * (len(s) + 1)
128 for i, d in enumerate(devices):
129 p = torch.cuda.get_device_properties(i)
130 s += f"{'' if i == 0 else space}CUDA:{d} ({p.name}, {p.total_memory / (1 << 20):.0f}MiB)\n" # bytes to MB
131 arg = "cuda:0"
132 elif mps and getattr(torch, "has_mps", False) and torch.backends.mps.is_available(): # prefer MPS if available
133 s += "MPS\n"
134 arg = "mps"
135 else: # revert to CPU
136 s += "CPU\n"
137 arg = "cpu"
138
139 if not newline:
140 s = s.rstrip()
141 LOGGER.info(s)
142 return torch.device(arg)

152def profile(input, ops, n=10, device=None):
153 """YOLOv5 speed/memory/FLOPs profiler
154 Usage:
155 input = torch.randn(16, 3, 640, 640)
156 m1 = lambda x: x * torch.sigmoid(x)
157 m2 = nn.SiLU()
158 profile(input, [m1, m2], n=100) # profile over 100 iterations
159 """
160 results = []
161 if not isinstance(device, torch.device):
162 device = select_device(device)
163 print(
164 f"{'Params':>12s}{'GFLOPs':>12s}{'GPU_mem (GB)':>14s}{'forward (ms)':>14s}{'backward (ms)':>14s}"
165 f"{'input':>24s}{'output':>24s}"
166 )
167
168 for x in input if isinstance(input, list) else [input]:
169 x = x.to(device)
170 x.requires_grad = True
171 for m in ops if isinstance(ops, list) else [ops]:
172 m = m.to(device) if hasattr(m, "to") else m # device
173 m = m.half() if hasattr(m, "half") and isinstance(x, torch.Tensor) and x.dtype is torch.float16 else m
174 tf, tb, t = 0, 0, [0, 0, 0] # dt forward, backward
175 try:
176 flops = thop.profile(m, inputs=(x,), verbose=False)[0] / 1e9 * 2 # GFLOPs
177 except Exception:
178 flops = 0
179
180 try:
181 for _ in range(n):
182 t[0] = time_sync()
183 y = m(x)
184 t[1] = time_sync()
185 try:
186 _ = (sum(yi.sum() for yi in y) if isinstance(y, list) else y).sum().backward()
187 t[2] = time_sync()
188 except Exception: # no backward method
189 # print(e) # for debug
190 t[2] = float("nan")
191 tf += (t[1] - t[0]) * 1000 / n # ms per op forward
192 tb += (t[2] - t[1]) * 1000 / n # ms per op backward
193 mem = torch.cuda.memory_reserved() / 1e9 if torch.cuda.is_available() else 0 # (GB)
194 s_in, s_out = (tuple(x.shape) if isinstance(x, torch.Tensor) else "list" for x in (x, y)) # shapes
195 p = sum(x.numel() for x in m.parameters()) if isinstance(m, nn.Module) else 0 # parameters
196 print(f"{p:12}{flops:12.4g}{mem:>14.3f}{tf:14.4g}{tb:14.4g}{str(s_in):>24s}{str(s_out):>24s}")
197 results.append([p, flops, mem, tf, tb, s_in, s_out])
198 except Exception as e:
199 print(e)
200 results.append(None)
201 torch.cuda.empty_cache()
202 return results

252def fuse_conv_and_bn(conv, bn):
253 # Fuse Conv2d() and BatchNorm2d() layers https://tehnokv.com/posts/fusing-batchnorm-and-conv/
254 fusedconv = (
255 nn.Conv2d(
256 conv.in_channels,
257 conv.out_channels,
258 kernel_size=conv.kernel_size,
259 stride=conv.stride,
260 padding=conv.padding,
261 dilation=conv.dilation,
262 groups=conv.groups,
263 bias=True,
264 )
265 .requires_grad_(False)
266 .to(conv.weight.device)
267 )
268
269 # Prepare filters
270 w_conv = conv.weight.clone().view(conv.out_channels, -1)
271 w_bn = torch.diag(bn.weight.div(torch.sqrt(bn.eps + bn.running_var)))
272 fusedconv.weight.copy_(torch.mm(w_bn, w_conv).view(fusedconv.weight.shape))
273
274 # Prepare spatial bias
275 b_conv = torch.zeros(conv.weight.size(0), device=conv.weight.device) if conv.bias is None else conv.bias
276 b_bn = bn.bias - bn.weight.mul(bn.running_mean).div(torch.sqrt(bn.running_var + bn.eps))
277 fusedconv.bias.copy_(torch.mm(w_bn, b_conv.reshape(-1, 1)).reshape(-1) + b_bn)
278
279 return fusedconv

309def scale_img(img, ratio=1.0, same_shape=False, gs=32): # img(16,3,256,416)
310 # Scales img(bs,3,y,x) by ratio constrained to gs-multiple
311 if ratio == 1.0:
312 return img
313 h, w = img.shape[2:]
314 s = (int(h * ratio), int(w * ratio)) # new size
315 img = F.interpolate(img, size=s, mode="bilinear", align_corners=False) # resize
316 if not same_shape: # pad/crop img
317 h, w = (math.ceil(x * ratio / gs) * gs for x in (h, w))
318 return F.pad(img, [0, w - s[1], 0, h - s[0]], value=0.447) # value = imagenet mean

330def smart_optimizer(model, name="Adam", lr=0.001, momentum=0.9, decay=1e-5):
331 # YOLOv5 3-param group optimizer: 0) weights with decay, 1) weights no decay, 2) biases no decay
332 g = [], [], [] # optimizer parameter groups
333 bn = tuple(v for k, v in nn.__dict__.items() if "Norm" in k) # normalization layers, i.e. BatchNorm2d()
334 for v in model.modules():
335 for p_name, p in v.named_parameters(recurse=0):
336 if p_name == "bias": # bias (no decay)
337 g[2].append(p)
338 elif p_name == "weight" and isinstance(v, bn): # weight (no decay)
339 g[1].append(p)
340 else:
341 g[0].append(p) # weight (with decay)
342
343 if name == "Adam":
344 optimizer = torch.optim.Adam(g[2], lr=lr, betas=(momentum, 0.999)) # adjust beta1 to momentum
345 elif name == "AdamW":
346 optimizer = torch.optim.AdamW(g[2], lr=lr, betas=(momentum, 0.999), weight_decay=0.0)
347 elif name == "RMSProp":
348 optimizer = torch.optim.RMSprop(g[2], lr=lr, momentum=momentum)
349 elif name == "SGD":
350 optimizer = torch.optim.SGD(g[2], lr=lr, momentum=momentum, nesterov=True)
351 else:
352 raise NotImplementedError(f"Optimizer {name} not implemented.")
353
354 optimizer.add_param_group({"params": g[0], "weight_decay": decay}) # add g0 with weight_decay
355 optimizer.add_param_group({"params": g[1], "weight_decay": 0.0}) # add g1 (BatchNorm2d weights)
356 LOGGER.info(
357 f"{colorstr('optimizer:')} {type(optimizer).__name__}(lr={lr}) with parameter groups "
358 f'{len(g[1])} weight(decay=0.0), {len(g[0])} weight(decay={decay}), {len(g[2])} bias'
359 )
360 return optimizer

375def smart_resume(ckpt, optimizer, ema=None, weights="yolov5s.pt", epochs=300, resume=True):
376 # Resume training from a partially trained checkpoint
377 best_fitness = 0.0
378 start_epoch = ckpt["epoch"] + 1
379 if ckpt["optimizer"] is not None:
380 optimizer.load_state_dict(ckpt["optimizer"]) # optimizer
381 best_fitness = ckpt["best_fitness"]
382 if ema and ckpt.get("ema"):
383 ema.ema.load_state_dict(ckpt["ema"].float().state_dict()) # EMA
384 ema.updates = ckpt["updates"]
385 if resume:
386 assert start_epoch > 0, (
387 f"{weights} training to {epochs} epochs is finished, nothing to resume.\n"
388 f"Start a new training without --resume, i.e. 'python train.py --weights {weights}'"
389 )
390 LOGGER.info(f"Resuming training from {weights} from epoch {start_epoch} to {epochs} total epochs")
391 if epochs < start_epoch:
392 LOGGER.info(f"{weights} has been trained for {ckpt['epoch']} epochs. Fine-tuning for {epochs} more epochs.")
393 epochs += ckpt["epoch"] # finetune additional epochs
394 return best_fitness, start_epoch, epochs

397class EarlyStopping:
398 # YOLOv5 simple early stopper
399 def __init__(self, patience=30):
400 self.best_fitness = 0.0 # i.e. mAP
401 self.best_epoch = 0
402 self.patience = patience or float("inf") # epochs to wait after fitness stops improving to stop
403 self.possible_stop = False # possible stop may occur next epoch
404
405 def __call__(self, epoch, fitness):
406 if fitness >= self.best_fitness: # >= 0 to allow for early zero-fitness stage of training
407 self.best_epoch = epoch
408 self.best_fitness = fitness
409 delta = epoch - self.best_epoch # epochs without improvement
410 self.possible_stop = delta >= (self.patience - 1) # possible stop may occur next epoch
411 stop = delta >= self.patience # stop training if patience exceeded
412 if stop:
413 LOGGER.info(
414 f"Stopping training early as no improvement observed in last {self.patience} epochs. "
415 f"Best results observed at epoch {self.best_epoch}, best model saved as best.pt.\n"
416 f"To update EarlyStopping(patience={self.patience}) pass a new patience value, "
417 f"i.e. `python train.py --patience 300` or use `--patience 0` to disable EarlyStopping."
418 )
419 return stop

422class ModelEMA:
423 """Updated Exponential Moving Average (EMA) from https://github.com/rwightman/pytorch-image-models
424 Keeps a moving average of everything in the model state_dict (parameters and buffers)
425 For EMA details see https://www.tensorflow.org/api_docs/python/tf/train/ExponentialMovingAverage
426 """
427
428 def __init__(self, model, decay=0.9999, tau=2000, updates=0):
429 # Create EMA
430 self.ema = deepcopy(de_parallel(model)).eval() # FP32 EMA
431 self.updates = updates # number of EMA updates
432 self.decay = lambda x: decay * (1 - math.exp(-x / tau)) # decay exponential ramp (to help early epochs)
433 for p in self.ema.parameters():
434 p.requires_grad_(False)
435
436 def update(self, model):
437 # Update EMA parameters
438 self.updates += 1
439 d = self.decay(self.updates)
440
441 msd = de_parallel(model).state_dict() # model state_dict
442 for k, v in self.ema.state_dict().items():
443 if v.dtype.is_floating_point: # true for FP16 and FP32
444 v *= d
445 v += (1 - d) * msd[k].detach()
446 # assert v.dtype == msd[k].dtype == torch.float32, f'{k}: EMA {v.dtype} and model {msd[k].dtype} must be FP32'
447
448 def update_attr(self, model, include=(), exclude=("process_group", "reducer")):
449 # Update EMA attributes
450 copy_attr(self.ema, model, include, exclude)
