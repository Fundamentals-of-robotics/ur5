59def _json_default(value):
60 """
61 Format `value` for JSON serialization (e.g. unwrap tensors).
62
63 Fall back to strings.
64 """
65 if isinstance(value, torch.Tensor):
66 try:
67 value = value.item()
68 except ValueError: # "only one element tensors can be converted to Python scalars"
69 pass
70 return value if isinstance(value, float) else str(value)

240 def on_fit_epoch_end(self, vals, epoch, best_fitness, fi):
241 # Callback runs at the end of each fit (train+val) epoch
242 x = dict(zip(self.keys, vals))
243 if self.csv:
244 file = self.save_dir / "results.csv"
245 n = len(x) + 1 # number of cols
246 s = "" if file.exists() else (("%20s," * n % tuple(["epoch"] + self.keys)).rstrip(",") + "\n") # add header
247 with open(file, "a") as f:
248 f.write(s + ("%20.5g," * n % tuple([epoch] + vals)).rstrip(",") + "\n")
249 if self.ndjson_console or self.ndjson_file:
250 json_data = json.dumps(dict(epoch=epoch, **x), default=_json_default)
251 if self.ndjson_console:
252 print(json_data)
253 if self.ndjson_file:
254 file = self.save_dir / "results.ndjson"
255 with open(file, "a") as f:
256 print(json_data, file=f)
257
258 if self.tb:
259 for k, v in x.items():
260 self.tb.add_scalar(k, v, epoch)
261 elif self.clearml: # log to ClearML if TensorBoard not used
262 self.clearml.log_scalars(x, epoch)
263
264 if self.wandb:
265 if best_fitness == fi:
266 best_results = [epoch] + vals[3:7]
267 for i, name in enumerate(self.best_keys):
268 self.wandb.wandb_run.summary[name] = best_results[i] # log best results in the summary
269 self.wandb.log(x)
270 self.wandb.end_epoch()
271
272 if self.clearml:
273 self.clearml.current_epoch_logged_images = set() # reset epoch image limit
274 self.clearml.current_epoch += 1
275
276 if self.comet_logger:
277 self.comet_logger.on_fit_epoch_end(x, epoch=epoch)

444def log_tensorboard_graph(tb, model, imgsz=(640, 640)):
445 # Log model graph to TensorBoard
446 try:
447 p = next(model.parameters()) # for device, type
448 imgsz = (imgsz, imgsz) if isinstance(imgsz, int) else imgsz # expand
449 im = torch.zeros((1, 3, *imgsz)).to(p.device).type_as(p) # input image (WARNING: must be zeros, not empty)
450 with warnings.catch_warnings():
451 warnings.simplefilter("ignore") # suppress jit trace warning
452 tb.add_graph(torch.jit.trace(de_parallel(model), im, strict=False), [])
453 except Exception as e:
454 LOGGER.warning(f"WARNING ⚠️ TensorBoard graph visualization failure {e}")
