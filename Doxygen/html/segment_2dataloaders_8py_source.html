232 def load_mosaic(self, index):
233 # YOLOv5 4-mosaic loader. Loads 1 image + 3 random images into a 4-image mosaic
234 labels4, segments4 = [], []
235 s = self.img_size
236 yc, xc = (int(random.uniform(-x, 2 * s + x)) for x in self.mosaic_border) # mosaic center x, y
237
238 # 3 additional image indices
239 indices = [index] + random.choices(self.indices, k=3) # 3 additional image indices
240 for i, index in enumerate(indices):
241 # Load image
242 img, _, (h, w) = self.load_image(index)
243
244 # place img in img4
245 if i == 0: # top left
246 img4 = np.full((s * 2, s * 2, img.shape[2]), 114, dtype=np.uint8) # base image with 4 tiles
247 x1a, y1a, x2a, y2a = max(xc - w, 0), max(yc - h, 0), xc, yc # xmin, ymin, xmax, ymax (large image)
248 x1b, y1b, x2b, y2b = w - (x2a - x1a), h - (y2a - y1a), w, h # xmin, ymin, xmax, ymax (small image)
249 elif i == 1: # top right
250 x1a, y1a, x2a, y2a = xc, max(yc - h, 0), min(xc + w, s * 2), yc
251 x1b, y1b, x2b, y2b = 0, h - (y2a - y1a), min(w, x2a - x1a), h
252 elif i == 2: # bottom left
253 x1a, y1a, x2a, y2a = max(xc - w, 0), yc, xc, min(s * 2, yc + h)
254 x1b, y1b, x2b, y2b = w - (x2a - x1a), 0, w, min(y2a - y1a, h)
255 elif i == 3: # bottom right
256 x1a, y1a, x2a, y2a = xc, yc, min(xc + w, s * 2), min(s * 2, yc + h)
257 x1b, y1b, x2b, y2b = 0, 0, min(w, x2a - x1a), min(y2a - y1a, h)
258
259 img4[y1a:y2a, x1a:x2a] = img[y1b:y2b, x1b:x2b] # img4[ymin:ymax, xmin:xmax]
260 padw = x1a - x1b
261 padh = y1a - y1b
262
263 labels, segments = self.labels[index].copy(), self.segments[index].copy()
264
265 if labels.size:
266 labels[:, 1:] = xywhn2xyxy(labels[:, 1:], w, h, padw, padh) # normalized xywh to pixel xyxy format
267 segments = [xyn2xy(x, w, h, padw, padh) for x in segments]
268 labels4.append(labels)
269 segments4.extend(segments)
270
271 # Concat/clip labels
272 labels4 = np.concatenate(labels4, 0)
273 for x in (labels4[:, 1:], *segments4):
274 np.clip(x, 0, 2 * s, out=x) # clip when using random_perspective()
275 # img4, labels4 = replicate(img4, labels4) # replicate
276
277 # Augment
278 img4, labels4, segments4 = copy_paste(img4, labels4, segments4, p=self.hyp["copy_paste"])
279 img4, labels4, segments4 = random_perspective(
280 img4,
281 labels4,
282 segments4,
283 degrees=self.hyp["degrees"],
284 translate=self.hyp["translate"],
285 scale=self.hyp["scale"],
286 shear=self.hyp["shear"],
287 perspective=self.hyp["perspective"],
288 border=self.mosaic_border,
289 ) # border to remove
290 return img4, labels4, segments4

301def polygon2mask(img_size, polygons, color=1, downsample_ratio=1):
302 """
303 Args:
304 img_size (tuple): The image size.
305 polygons (np.ndarray): [N, M], N is the number of polygons,
306 M is the number of points(Be divided by 2).
307 """
308 mask = np.zeros(img_size, dtype=np.uint8)
309 polygons = np.asarray(polygons)
310 polygons = polygons.astype(np.int32)
311 shape = polygons.shape
312 polygons = polygons.reshape(shape[0], -1, 2)
313 cv2.fillPoly(mask, polygons, color=color)
314 nh, nw = (img_size[0] // downsample_ratio, img_size[1] // downsample_ratio)
315 # NOTE: fillPoly firstly then resize is trying the keep the same way
316 # of loss calculation when mask-ratio=1.
317 mask = cv2.resize(mask, (nw, nh))
318 return mask

321def polygons2masks(img_size, polygons, color, downsample_ratio=1):
322 """
323 Args:
324 img_size (tuple): The image size.
325 polygons (list[np.ndarray]): each polygon is [N, M],
326 N is the number of polygons,
327 M is the number of points(Be divided by 2).
328 """
329 masks = []
330 for si in range(len(polygons)):
331 mask = polygon2mask(img_size, [polygons[si].reshape(-1)], color, downsample_ratio)
332 masks.append(mask)
333 return np.array(masks)

336def polygons2masks_overlap(img_size, segments, downsample_ratio=1):
337 """Return a (640, 640) overlap mask."""
338 masks = np.zeros(
339 (img_size[0] // downsample_ratio, img_size[1] // downsample_ratio),
340 dtype=np.int32 if len(segments) > 255 else np.uint8,
341 )
342 areas = []
343 ms = []
344 for si in range(len(segments)):
345 mask = polygon2mask(
346 img_size,
347 [segments[si].reshape(-1)],
348 downsample_ratio=downsample_ratio,
349 color=1,
350 )
351 ms.append(mask)
352 areas.append(mask.sum())
353 areas = np.asarray(areas)
354 index = np.argsort(-areas)
355 ms = np.array(ms)[index]
356 for i in range(len(segments)):
357 mask = ms[i] * (i + 1)
358 masks = masks + mask
359 masks = np.clip(masks, a_min=0, a_max=i + 1)
360 return masks, index
