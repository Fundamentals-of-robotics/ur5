72 source=ROOT / "data/images", # file/dir/URL/glob/screen/0(webcam)
73 data=ROOT / "data/coco128.yaml", # dataset.yaml path
74 imgsz=(640, 640), # inference size (height, width)
75 conf_thres=0.25, # confidence threshold
76 iou_thres=0.45, # NMS IOU threshold
77 max_det=1000, # maximum detections per image
78 device="", # cuda device, i.e. 0 or 0,1,2,3 or cpu
79 view_img=False, # show results
80 save_txt=False, # save results to *.txt
81 save_conf=False, # save confidences in --save-txt labels
82 save_crop=False, # save cropped prediction boxes
83 nosave=False, # do not save images/videos
84 classes=None, # filter by class: --class 0, or --class 0 2 3
85 agnostic_nms=False, # class-agnostic NMS
86 augment=False, # augmented inference
87 visualize=False, # visualize features
88 update=False, # update all models
89 project=ROOT / "runs/predict-seg", # save results to project/name
90 name="exp", # save results to project/name
91 exist_ok=False, # existing project/name ok, do not increment
92 line_thickness=3, # bounding box thickness (pixels)
93 hide_labels=False, # hide labels
94 hide_conf=False, # hide confidences
95 half=False, # use FP16 half-precision inference
96 dnn=False, # use OpenCV DNN for ONNX inference
97 vid_stride=1, # video frame-rate stride
98 retina_masks=False,
99):
100 source = str(source)
101 save_img = not nosave and not source.endswith(".txt") # save inference images
102 is_file = Path(source).suffix[1:] in (IMG_FORMATS + VID_FORMATS)
103 is_url = source.lower().startswith(("rtsp://", "rtmp://", "http://", "https://"))
104 webcam = source.isnumeric() or source.endswith(".streams") or (is_url and not is_file)
105 screenshot = source.lower().startswith("screen")
106 if is_url and is_file:
107 source = check_file(source) # download
108
109 # Directories
110 save_dir = increment_path(Path(project) / name, exist_ok=exist_ok) # increment run
111 (save_dir / "labels" if save_txt else save_dir).mkdir(parents=True, exist_ok=True) # make dir
112
113 # Load model
114 device = select_device(device)
115 model = DetectMultiBackend(weights, device=device, dnn=dnn, data=data, fp16=half)
116 stride, names, pt = model.stride, model.names, model.pt
117 imgsz = check_img_size(imgsz, s=stride) # check image size
118
119 # Dataloader
120 bs = 1 # batch_size
121 if webcam:
122 view_img = check_imshow(warn=True)
123 dataset = LoadStreams(source, img_size=imgsz, stride=stride, auto=pt, vid_stride=vid_stride)
124 bs = len(dataset)
125 elif screenshot:
126 dataset = LoadScreenshots(source, img_size=imgsz, stride=stride, auto=pt)
127 else:
128 dataset = LoadImages(source, img_size=imgsz, stride=stride, auto=pt, vid_stride=vid_stride)
129 vid_path, vid_writer = [None] * bs, [None] * bs
130
131 # Run inference
132 model.warmup(imgsz=(1 if pt else bs, 3, *imgsz)) # warmup
133 seen, windows, dt = 0, [], (Profile(device=device), Profile(device=device), Profile(device=device))
134 for path, im, im0s, vid_cap, s in dataset:
135 with dt[0]:
136 im = torch.from_numpy(im).to(model.device)
137 im = im.half() if model.fp16 else im.float() # uint8 to fp16/32
138 im /= 255 # 0 - 255 to 0.0 - 1.0
139 if len(im.shape) == 3:
140 im = im[None] # expand for batch dim
141
142 # Inference
143 with dt[1]:
144 visualize = increment_path(save_dir / Path(path).stem, mkdir=True) if visualize else False
145 pred, proto = model(im, augment=augment, visualize=visualize)[:2]
146
147 # NMS
148 with dt[2]:
149 pred = non_max_suppression(pred, conf_thres, iou_thres, classes, agnostic_nms, max_det=max_det, nm=32)
150
151 # Second-stage classifier (optional)
152 # pred = utils.general.apply_classifier(pred, classifier_model, im, im0s)
153
154 # Process predictions
155 for i, det in enumerate(pred): # per image
156 seen += 1
157 if webcam: # batch_size >= 1
158 p, im0, frame = path[i], im0s[i].copy(), dataset.count
159 s += f"{i}: "
160 else:
161 p, im0, frame = path, im0s.copy(), getattr(dataset, "frame", 0)
162
163 p = Path(p) # to Path
164 save_path = str(save_dir / p.name) # im.jpg
165 txt_path = str(save_dir / "labels" / p.stem) + ("" if dataset.mode == "image" else f"_{frame}") # im.txt
166 s += "%gx%g " % im.shape[2:] # print string
167 imc = im0.copy() if save_crop else im0 # for save_crop
168 annotator = Annotator(im0, line_width=line_thickness, example=str(names))
169 if len(det):
170 if retina_masks:
171 # scale bbox first the crop masks
172 det[:, :4] = scale_boxes(im.shape[2:], det[:, :4], im0.shape).round() # rescale boxes to im0 size
173 masks = process_mask_native(proto[i], det[:, 6:], det[:, :4], im0.shape[:2]) # HWC
174 else:
175 masks = process_mask(proto[i], det[:, 6:], det[:, :4], im.shape[2:], upsample=True) # HWC
176 det[:, :4] = scale_boxes(im.shape[2:], det[:, :4], im0.shape).round() # rescale boxes to im0 size
177
178 # Segments
179 if save_txt:
180 segments = [
181 scale_segments(im0.shape if retina_masks else im.shape[2:], x, im0.shape, normalize=True)
182 for x in reversed(masks2segments(masks))
183 ]
184
185 # Print results
186 for c in det[:, 5].unique():
187 n = (det[:, 5] == c).sum() # detections per class
188 s += f"{n} {names[int(c)]}{'s' * (n > 1)}, " # add to string
189
190 # Mask plotting
191 annotator.masks(
192 masks,
193 colors=[colors(x, True) for x in det[:, 5]],
194 im_gpu=torch.as_tensor(im0, dtype=torch.float16).to(device).permute(2, 0, 1).flip(0).contiguous()
195 / 255
196 if retina_masks
197 else im[i],
198 )
199
200 # Write results
201 for j, (*xyxy, conf, cls) in enumerate(reversed(det[:, :6])):
202 if save_txt: # Write to file
203 seg = segments[j].reshape(-1) # (n,2) to (n*2)
204 line = (cls, *seg, conf) if save_conf else (cls, *seg) # label format
205 with open(f"{txt_path}.txt", "a") as f:
206 f.write(("%g " * len(line)).rstrip() % line + "\n")
207
208 if save_img or save_crop or view_img: # Add bbox to image
209 c = int(cls) # integer class
210 label = None if hide_labels else (names[c] if hide_conf else f"{names[c]} {conf:.2f}")
211 annotator.box_label(xyxy, label, color=colors(c, True))
212 # annotator.draw.polygon(segments[j], outline=colors(c, True), width=3)
213 if save_crop:
214 save_one_box(xyxy, imc, file=save_dir / "crops" / names[c] / f"{p.stem}.jpg", BGR=True)
215
216 # Stream results
217 im0 = annotator.result()
218 if view_img:
219 if platform.system() == "Linux" and p not in windows:
220 windows.append(p)
221 cv2.namedWindow(str(p), cv2.WINDOW_NORMAL | cv2.WINDOW_KEEPRATIO) # allow window resize (Linux)
222 cv2.resizeWindow(str(p), im0.shape[1], im0.shape[0])
223 cv2.imshow(str(p), im0)
224 if cv2.waitKey(1) == ord("q"): # 1 millisecond
225 exit()
226
227 # Save results (image with detections)
228 if save_img:
229 if dataset.mode == "image":
230 cv2.imwrite(save_path, im0)
231 else: # 'video' or 'stream'
232 if vid_path[i] != save_path: # new video
233 vid_path[i] = save_path
234 if isinstance(vid_writer[i], cv2.VideoWriter):
235 vid_writer[i].release() # release previous video writer
236 if vid_cap: # video
237 fps = vid_cap.get(cv2.CAP_PROP_FPS)
238 w = int(vid_cap.get(cv2.CAP_PROP_FRAME_WIDTH))
239 h = int(vid_cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
240 else: # stream
241 fps, w, h = 30, im0.shape[1], im0.shape[0]
242 save_path = str(Path(save_path).with_suffix(".mp4")) # force *.mp4 suffix on results videos
243 vid_writer[i] = cv2.VideoWriter(save_path, cv2.VideoWriter_fourcc(*"mp4v"), fps, (w, h))
244 vid_writer[i].write(im0)
245
246 # Print time (inference-only)
247 LOGGER.info(f"{s}{'' if len(det) else '(no detections), '}{dt[1].dt * 1E3:.1f}ms")
248
249 # Print results
250 t = tuple(x.t / seen * 1e3 for x in dt) # speeds per image
251 LOGGER.info(f"Speed: %.1fms pre-process, %.1fms inference, %.1fms NMS per image at shape {(1, 3, *imgsz)}" % t)
252 if save_txt or save_img:
253 s = f"\n{len(list(save_dir.glob('labels/*.txt')))} labels saved to {save_dir / 'labels'}" if save_txt else ""
254 LOGGER.info(f"Results saved to {colorstr('bold', save_dir)}{s}")
255 if update:
256 strip_optimizer(weights[0]) # update model (to fix SourceChangeWarning)
