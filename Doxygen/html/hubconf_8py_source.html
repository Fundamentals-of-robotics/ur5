16def _create(name, pretrained=True, channels=3, classes=80, autoshape=True, verbose=True, device=None):
17 """
18 Creates or loads a YOLOv5 model.
19
20 Arguments:
21 name (str): model name 'yolov5s' or path 'path/to/best.pt'
22 pretrained (bool): load pretrained weights into the model
23 channels (int): number of input channels
24 classes (int): number of model classes
25 autoshape (bool): apply YOLOv5 .autoshape() wrapper to model
26 verbose (bool): print all information to screen
27 device (str, torch.device, None): device to use for model parameters
28
29 Returns:
30 YOLOv5 model
31 """
32 from pathlib import Path
33
34 from models.common import AutoShape, DetectMultiBackend
35 from models.experimental import attempt_load
36 from models.yolo import ClassificationModel, DetectionModel, SegmentationModel
37 from utils.downloads import attempt_download
38 from utils.general import LOGGER, ROOT, check_requirements, intersect_dicts, logging
39 from utils.torch_utils import select_device
40
41 if not verbose:
42 LOGGER.setLevel(logging.WARNING)
43 check_requirements(ROOT / "requirements.txt", exclude=("opencv-python", "tensorboard", "thop"))
44 name = Path(name)
45 path = name.with_suffix(".pt") if name.suffix == "" and not name.is_dir() else name # checkpoint path
46 try:
47 device = select_device(device)
48 if pretrained and channels == 3 and classes == 80:
49 try:
50 model = DetectMultiBackend(path, device=device, fuse=autoshape) # detection model
51 if autoshape:
52 if model.pt and isinstance(model.model, ClassificationModel):
53 LOGGER.warning(
54 "WARNING ⚠️ YOLOv5 ClassificationModel is not yet AutoShape compatible. "
55 "You must pass torch tensors in BCHW to this model, i.e. shape(1,3,224,224)."
56 )
57 elif model.pt and isinstance(model.model, SegmentationModel):
58 LOGGER.warning(
59 "WARNING ⚠️ YOLOv5 SegmentationModel is not yet AutoShape compatible. "
60 "You will not be able to run inference with this model."
61 )
62 else:
63 model = AutoShape(model) # for file/URI/PIL/cv2/np inputs and NMS
64 except Exception:
65 model = attempt_load(path, device=device, fuse=False) # arbitrary model
66 else:
67 cfg = list((Path(__file__).parent / "models").rglob(f"{path.stem}.yaml"))[0] # model.yaml path
68 model = DetectionModel(cfg, channels, classes) # create model
69 if pretrained:
70 ckpt = torch.load(attempt_download(path), map_location=device) # load
71 csd = ckpt["model"].float().state_dict() # checkpoint state_dict as FP32
72 csd = intersect_dicts(csd, model.state_dict(), exclude=["anchors"]) # intersect
73 model.load_state_dict(csd, strict=False) # load
74 if len(ckpt["model"].names) == classes:
75 model.names = ckpt["model"].names # set class names attribute
76 if not verbose:
77 LOGGER.setLevel(logging.INFO) # reset to default
78 return model.to(device)
79
80 except Exception as e:
81 help_url = "https://docs.ultralytics.com/yolov5/tutorials/pytorch_hub_model_loading"
82 s = f"{e}. Cache may be out of date, try `force_reload=True` or see {help_url} for help."
83 raise Exception(s) from e
