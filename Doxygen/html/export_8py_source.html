93class iOSModel(torch.nn.Module):
94 def __init__(self, model, im):
95 super().__init__()
96 b, c, h, w = im.shape # batch, channel, height, width
97 self.model = model
98 self.nc = model.nc # number of classes
99 if w == h:
100 self.normalize = 1.0 / w
101 else:
102 self.normalize = torch.tensor([1.0 / w, 1.0 / h, 1.0 / w, 1.0 / h]) # broadcast (slower, smaller)
103 # np = model(im)[0].shape[1] # number of points
104 # self.normalize = torch.tensor([1. / w, 1. / h, 1. / w, 1. / h]).expand(np, 4) # explicit (faster, larger)
105
106 def forward(self, x):
107 xywh, conf, cls = self.model(x)[0].squeeze().split((4, 1, self.nc), 1)
108 return cls * conf, xywh * self.normalize # confidence (3780, 80), coordinates (3780, 4)

130def try_export(inner_func):
131 # YOLOv5 export decorator, i..e @try_export
132 inner_args = get_default_args(inner_func)
133
134 def outer_func(*args, **kwargs):
135 prefix = inner_args["prefix"]
136 try:
137 with Profile() as dt:
138 f, model = inner_func(*args, **kwargs)
139 LOGGER.info(f"{prefix} export success ‚úÖ {dt.t:.1f}s, saved as {f} ({file_size(f):.1f} MB)")
140 return f, model
141 except Exception as e:
142 LOGGER.info(f"{prefix} export failure ‚ùå {dt.t:.1f}s: {e}")
143 return None, None
144
145 return outer_func

149def export_torchscript(model, im, file, optimize, prefix=colorstr("TorchScript:")):
150 # YOLOv5 TorchScript model export
151 LOGGER.info(f"\n{prefix} starting export with torch {torch.__version__}...")
152 f = file.with_suffix(".torchscript")
153
154 ts = torch.jit.trace(model, im, strict=False)
155 d = {"shape": im.shape, "stride": int(max(model.stride)), "names": model.names}
156 extra_files = {"config.txt": json.dumps(d)} # torch._C.ExtraFilesMap()
157 if optimize: # https://pytorch.org/tutorials/recipes/mobile_interpreter.html
158 optimize_for_mobile(ts)._save_for_lite_interpreter(str(f), _extra_files=extra_files)
159 else:
160 ts.save(str(f), _extra_files=extra_files)
161 return f, None

164@try_export
165def export_onnx(model, im, file, opset, dynamic, simplify, prefix=colorstr("ONNX:")):
166 # YOLOv5 ONNX export
167 check_requirements("onnx>=1.12.0")
168 import onnx
169
170 LOGGER.info(f"\n{prefix} starting export with onnx {onnx.__version__}...")
171 f = str(file.with_suffix(".onnx"))
172
173 output_names = ["output0", "output1"] if isinstance(model, SegmentationModel) else ["output0"]
174 if dynamic:
175 dynamic = {"images": {0: "batch", 2: "height", 3: "width"}} # shape(1,3,640,640)
176 if isinstance(model, SegmentationModel):
177 dynamic["output0"] = {0: "batch", 1: "anchors"} # shape(1,25200,85)
178 dynamic["output1"] = {0: "batch", 2: "mask_height", 3: "mask_width"} # shape(1,32,160,160)
179 elif isinstance(model, DetectionModel):
180 dynamic["output0"] = {0: "batch", 1: "anchors"} # shape(1,25200,85)
181
182 torch.onnx.export(
183 model.cpu() if dynamic else model, # --dynamic only compatible with cpu
184 im.cpu() if dynamic else im,
185 f,
186 verbose=False,
187 opset_version=opset,
188 do_constant_folding=True, # WARNING: DNN inference with torch>=1.12 may require do_constant_folding=False
189 input_names=["images"],
190 output_names=output_names,
191 dynamic_axes=dynamic or None,
192 )
193
194 # Checks
195 model_onnx = onnx.load(f) # load onnx model
196 onnx.checker.check_model(model_onnx) # check onnx model
197
198 # Metadata
199 d = {"stride": int(max(model.stride)), "names": model.names}
200 for k, v in d.items():
201 meta = model_onnx.metadata_props.add()
202 meta.key, meta.value = k, str(v)
203 onnx.save(model_onnx, f)
204
205 # Simplify
206 if simplify:
207 try:
208 cuda = torch.cuda.is_available()
209 check_requirements(("onnxruntime-gpu" if cuda else "onnxruntime", "onnx-simplifier>=0.4.1"))
210 import onnxsim
211
212 LOGGER.info(f"{prefix} simplifying with onnx-simplifier {onnxsim.__version__}...")
213 model_onnx, check = onnxsim.simplify(model_onnx)
214 assert check, "assert check failed"
215 onnx.save(model_onnx, f)
216 except Exception as e:
217 LOGGER.info(f"{prefix} simplifier failure: {e}")
218 return f, model_onnx

252 def transform_fn(data_item):
253 """
254 Quantization transform function.
255
256 Extracts and preprocess input data from dataloader item for quantization.
257 Parameters:
258 data_item: Tuple with data item produced by DataLoader during iteration
259 Returns:
260 input_tensor: Input data for quantization
261 """
262 assert data_item[0].dtype == torch.uint8, "input image must be uint8 for the quantization preprocessing"
263
264 img = data_item[0].numpy().astype(np.float32) # uint8 to fp16/32
265 img /= 255.0 # 0 - 255 to 0.0 - 1.0
266 return np.expand_dims(img, 0) if img.ndim == 3 else img
267
268 ds = gen_dataloader(data)
269 quantization_dataset = nncf.Dataset(ds, transform_fn)
270 ov_model = nncf.quantize(ov_model, quantization_dataset, preset=nncf.QuantizationPreset.MIXED)
271
272 ov.serialize(ov_model, f_ov) # save
273 yaml_save(Path(f) / file.with_suffix(".yaml").name, metadata) # add metadata.yaml
274 return f, None

317@try_export
318def export_engine(model, im, file, half, dynamic, simplify, workspace=4, verbose=False, prefix=colorstr("TensorRT:")):
319 # YOLOv5 TensorRT export https://developer.nvidia.com/tensorrt
320 assert im.device.type != "cpu", "export running on CPU but must be on GPU, i.e. `python export.py --device 0`"
321 try:
322 import tensorrt as trt
323 except Exception:
324 if platform.system() == "Linux":
325 check_requirements("nvidia-tensorrt", cmds="-U --index-url https://pypi.ngc.nvidia.com")
326 import tensorrt as trt
327
328 if trt.__version__[0] == "7": # TensorRT 7 handling https://github.com/ultralytics/yolov5/issues/6012
329 grid = model.model[-1].anchor_grid
330 model.model[-1].anchor_grid = [a[..., :1, :1, :] for a in grid]
331 export_onnx(model, im, file, 12, dynamic, simplify) # opset 12
332 model.model[-1].anchor_grid = grid
333 else: # TensorRT >= 8
334 check_version(trt.__version__, "8.0.0", hard=True) # require tensorrt>=8.0.0
335 export_onnx(model, im, file, 12, dynamic, simplify) # opset 12
336 onnx = file.with_suffix(".onnx")
337
338 LOGGER.info(f"\n{prefix} starting export with TensorRT {trt.__version__}...")
339 assert onnx.exists(), f"failed to export ONNX file: {onnx}"
340 f = file.with_suffix(".engine") # TensorRT engine file
341 logger = trt.Logger(trt.Logger.INFO)
342 if verbose:
343 logger.min_severity = trt.Logger.Severity.VERBOSE
344
345 builder = trt.Builder(logger)
346 config = builder.create_builder_config()
347 config.max_workspace_size = workspace * 1 << 30
348 # config.set_memory_pool_limit(trt.MemoryPoolType.WORKSPACE, workspace << 30) # fix TRT 8.4 deprecation notice
349
350 flag = 1 << int(trt.NetworkDefinitionCreationFlag.EXPLICIT_BATCH)
351 network = builder.create_network(flag)
352 parser = trt.OnnxParser(network, logger)
353 if not parser.parse_from_file(str(onnx)):
354 raise RuntimeError(f"failed to load ONNX file: {onnx}")
355
356 inputs = [network.get_input(i) for i in range(network.num_inputs)]
357 outputs = [network.get_output(i) for i in range(network.num_outputs)]
358 for inp in inputs:
359 LOGGER.info(f'{prefix} input "{inp.name}" with shape{inp.shape} {inp.dtype}')
360 for out in outputs:
361 LOGGER.info(f'{prefix} output "{out.name}" with shape{out.shape} {out.dtype}')
362
363 if dynamic:
364 if im.shape[0] <= 1:
365 LOGGER.warning(f"{prefix} WARNING ‚ö†Ô∏è --dynamic model requires maximum --batch-size argument")
366 profile = builder.create_optimization_profile()
367 for inp in inputs:
368 profile.set_shape(inp.name, (1, *im.shape[1:]), (max(1, im.shape[0] // 2), *im.shape[1:]), im.shape)
369 config.add_optimization_profile(profile)
370
371 LOGGER.info(f"{prefix} building FP{16 if builder.platform_has_fast_fp16 and half else 32} engine as {f}")
372 if builder.platform_has_fast_fp16 and half:
373 config.set_flag(trt.BuilderFlag.FP16)
374 with builder.build_engine(network, config) as engine, open(f, "wb") as t:
375 t.write(engine.serialize())
376 return f, None

379@try_export
380def export_saved_model(
381 model,
382 im,
383 file,
384 dynamic,
385 tf_nms=False,
386 agnostic_nms=False,
387 topk_per_class=100,
388 topk_all=100,
389 iou_thres=0.45,
390 conf_thres=0.25,
391 keras=False,
392 prefix=colorstr("TensorFlow SavedModel:"),
393):
394 # YOLOv5 TensorFlow SavedModel export
395 try:
396 import tensorflow as tf
397 except Exception:
398 check_requirements(f"tensorflow{'' if torch.cuda.is_available() else '-macos' if MACOS else '-cpu'}")
399 import tensorflow as tf
400 from tensorflow.python.framework.convert_to_constants import convert_variables_to_constants_v2
401
402 from models.tf import TFModel
403
404 LOGGER.info(f"\n{prefix} starting export with tensorflow {tf.__version__}...")
405 if tf.__version__ > "2.13.1":
406 helper_url = "https://github.com/ultralytics/yolov5/issues/12489"
407 LOGGER.info(
408 f"WARNING ‚ö†Ô∏è using Tensorflow {tf.__version__} > 2.13.1 might cause issue when exporting the model to tflite {helper_url}"
409 ) # handling issue https://github.com/ultralytics/yolov5/issues/12489
410 f = str(file).replace(".pt", "_saved_model")
411 batch_size, ch, *imgsz = list(im.shape) # BCHW
412
413 tf_model = TFModel(cfg=model.yaml, model=model, nc=model.nc, imgsz=imgsz)
414 im = tf.zeros((batch_size, *imgsz, ch)) # BHWC order for TensorFlow
415 _ = tf_model.predict(im, tf_nms, agnostic_nms, topk_per_class, topk_all, iou_thres, conf_thres)
416 inputs = tf.keras.Input(shape=(*imgsz, ch), batch_size=None if dynamic else batch_size)
417 outputs = tf_model.predict(inputs, tf_nms, agnostic_nms, topk_per_class, topk_all, iou_thres, conf_thres)
418 keras_model = tf.keras.Model(inputs=inputs, outputs=outputs)
419 keras_model.trainable = False
420 keras_model.summary()
421 if keras:
422 keras_model.save(f, save_format="tf")
423 else:
424 spec = tf.TensorSpec(keras_model.inputs[0].shape, keras_model.inputs[0].dtype)
425 m = tf.function(lambda x: keras_model(x)) # full model
426 m = m.get_concrete_function(spec)
427 frozen_func = convert_variables_to_constants_v2(m)
428 tfm = tf.Module()
429 tfm.__call__ = tf.function(lambda x: frozen_func(x)[:4] if tf_nms else frozen_func(x), [spec])
430 tfm.__call__(im)
431 tf.saved_model.save(
432 tfm,
433 f,
434 options=tf.saved_model.SaveOptions(experimental_custom_gradients=False)
435 if check_version(tf.__version__, "2.6")
436 else tf.saved_model.SaveOptions(),
437 )
438 return f, keras_model

570def add_tflite_metadata(file, metadata, num_outputs):
571 # Add metadata to *.tflite models per https://www.tensorflow.org/lite/models/convert/metadata
572 with contextlib.suppress(ImportError):
573 # check_requirements('tflite_support')
574 from tflite_support import flatbuffers
575 from tflite_support import metadata as _metadata
576 from tflite_support import metadata_schema_py_generated as _metadata_fb
577
578 tmp_file = Path("/tmp/meta.txt")
579 with open(tmp_file, "w") as meta_f:
580 meta_f.write(str(metadata))
581
582 model_meta = _metadata_fb.ModelMetadataT()
583 label_file = _metadata_fb.AssociatedFileT()
584 label_file.name = tmp_file.name
585 model_meta.associatedFiles = [label_file]
586
587 subgraph = _metadata_fb.SubGraphMetadataT()
588 subgraph.inputTensorMetadata = [_metadata_fb.TensorMetadataT()]
589 subgraph.outputTensorMetadata = [_metadata_fb.TensorMetadataT()] * num_outputs
590 model_meta.subgraphMetadata = [subgraph]
591
592 b = flatbuffers.Builder(0)
593 b.Finish(model_meta.Pack(b), _metadata.MetadataPopulator.METADATA_FILE_IDENTIFIER)
594 metadata_buf = b.Output()
595
596 populator = _metadata.MetadataPopulator.with_model_file(file)
597 populator.load_metadata_buffer(metadata_buf)
598 populator.load_associated_files([str(tmp_file)])
599 populator.populate()
600 tmp_file.unlink()
