51class TFBN(keras.layers.Layer):
52 # TensorFlow BatchNormalization wrapper
53 def __init__(self, w=None):
54 super().__init__()
55 self.bn = keras.layers.BatchNormalization(
56 beta_initializer=keras.initializers.Constant(w.bias.numpy()),
57 gamma_initializer=keras.initializers.Constant(w.weight.numpy()),
58 moving_mean_initializer=keras.initializers.Constant(w.running_mean.numpy()),
59 moving_variance_initializer=keras.initializers.Constant(w.running_var.numpy()),
60 epsilon=w.eps,
61 )
62
63 def call(self, inputs):
64 return self.bn(inputs)

67class TFPad(keras.layers.Layer):
68 # Pad inputs in spatial dimensions 1 and 2
69 def __init__(self, pad):
70 super().__init__()
71 if isinstance(pad, int):
72 self.pad = tf.constant([[0, 0], [pad, pad], [pad, pad], [0, 0]])
73 else: # tuple/list
74 self.pad = tf.constant([[0, 0], [pad[0], pad[0]], [pad[1], pad[1]], [0, 0]])
75
76 def call(self, inputs):
77 return tf.pad(inputs, self.pad, mode="constant", constant_values=0)

80class TFConv(keras.layers.Layer):
81 # Standard convolution
82 def __init__(self, c1, c2, k=1, s=1, p=None, g=1, act=True, w=None):
83 # ch_in, ch_out, weights, kernel, stride, padding, groups
84 super().__init__()
85 assert g == 1, "TF v2.2 Conv2D does not support 'groups' argument"
86 # TensorFlow convolution padding is inconsistent with PyTorch (e.g. k=3 s=2 'SAME' padding)
87 # see https://stackoverflow.com/questions/52975843/comparing-conv2d-with-padding-between-tensorflow-and-pytorch
88 conv = keras.layers.Conv2D(
89 filters=c2,
90 kernel_size=k,
91 strides=s,
92 padding="SAME" if s == 1 else "VALID",
93 use_bias=not hasattr(w, "bn"),
94 kernel_initializer=keras.initializers.Constant(w.conv.weight.permute(2, 3, 1, 0).numpy()),
95 bias_initializer="zeros" if hasattr(w, "bn") else keras.initializers.Constant(w.conv.bias.numpy()),
96 )
97 self.conv = conv if s == 1 else keras.Sequential([TFPad(autopad(k, p)), conv])
98 self.bn = TFBN(w.bn) if hasattr(w, "bn") else tf.identity
99 self.act = activations(w.act) if act else tf.identity
100
101 def call(self, inputs):
102 return self.act(self.bn(self.conv(inputs)))

105class TFDWConv(keras.layers.Layer):
106 # Depthwise convolution
107 def __init__(self, c1, c2, k=1, s=1, p=None, act=True, w=None):
108 # ch_in, ch_out, weights, kernel, stride, padding, groups
109 super().__init__()
110 assert c2 % c1 == 0, f"TFDWConv() output={c2} must be a multiple of input={c1} channels"
111 conv = keras.layers.DepthwiseConv2D(
112 kernel_size=k,
113 depth_multiplier=c2 // c1,
114 strides=s,
115 padding="SAME" if s == 1 else "VALID",
116 use_bias=not hasattr(w, "bn"),
117 depthwise_initializer=keras.initializers.Constant(w.conv.weight.permute(2, 3, 1, 0).numpy()),
118 bias_initializer="zeros" if hasattr(w, "bn") else keras.initializers.Constant(w.conv.bias.numpy()),
119 )
120 self.conv = conv if s == 1 else keras.Sequential([TFPad(autopad(k, p)), conv])
121 self.bn = TFBN(w.bn) if hasattr(w, "bn") else tf.identity
122 self.act = activations(w.act) if act else tf.identity
123
124 def call(self, inputs):
125 return self.act(self.bn(self.conv(inputs)))

128class TFDWConvTranspose2d(keras.layers.Layer):
129 # Depthwise ConvTranspose2d
130 def __init__(self, c1, c2, k=1, s=1, p1=0, p2=0, w=None):
131 # ch_in, ch_out, weights, kernel, stride, padding, groups
132 super().__init__()
133 assert c1 == c2, f"TFDWConv() output={c2} must be equal to input={c1} channels"
134 assert k == 4 and p1 == 1, "TFDWConv() only valid for k=4 and p1=1"
135 weight, bias = w.weight.permute(2, 3, 1, 0).numpy(), w.bias.numpy()
136 self.c1 = c1
137 self.conv = [
138 keras.layers.Conv2DTranspose(
139 filters=1,
140 kernel_size=k,
141 strides=s,
142 padding="VALID",
143 output_padding=p2,
144 use_bias=True,
145 kernel_initializer=keras.initializers.Constant(weight[..., i : i + 1]),
146 bias_initializer=keras.initializers.Constant(bias[i]),
147 )
148 for i in range(c1)
149 ]
150
151 def call(self, inputs):
152 return tf.concat([m(x) for m, x in zip(self.conv, tf.split(inputs, self.c1, 3))], 3)[:, 1:-1, 1:-1]

155class TFFocus(keras.layers.Layer):
156 # Focus wh information into c-space
157 def __init__(self, c1, c2, k=1, s=1, p=None, g=1, act=True, w=None):
158 # ch_in, ch_out, kernel, stride, padding, groups
159 super().__init__()
160 self.conv = TFConv(c1 * 4, c2, k, s, p, g, act, w.conv)
161
162 def call(self, inputs): # x(b,w,h,c) -> y(b,w/2,h/2,4c)
163 # inputs = inputs / 255 # normalize 0-255 to 0-1
164 inputs = [inputs[:, ::2, ::2, :], inputs[:, 1::2, ::2, :], inputs[:, ::2, 1::2, :], inputs[:, 1::2, 1::2, :]]
165 return self.conv(tf.concat(inputs, 3))

168class TFBottleneck(keras.layers.Layer):
169 # Standard bottleneck
170 def __init__(self, c1, c2, shortcut=True, g=1, e=0.5, w=None): # ch_in, ch_out, shortcut, groups, expansion
171 super().__init__()
172 c_ = int(c2 * e) # hidden channels
173 self.cv1 = TFConv(c1, c_, 1, 1, w=w.cv1)
174 self.cv2 = TFConv(c_, c2, 3, 1, g=g, w=w.cv2)
175 self.add = shortcut and c1 == c2
176
177 def call(self, inputs):
178 return inputs + self.cv2(self.cv1(inputs)) if self.add else self.cv2(self.cv1(inputs))

213class TFBottleneckCSP(keras.layers.Layer):
214 # CSP Bottleneck https://github.com/WongKinYiu/CrossStagePartialNetworks
215 def __init__(self, c1, c2, n=1, shortcut=True, g=1, e=0.5, w=None):
216 # ch_in, ch_out, number, shortcut, groups, expansion
217 super().__init__()
218 c_ = int(c2 * e) # hidden channels
219 self.cv1 = TFConv(c1, c_, 1, 1, w=w.cv1)
220 self.cv2 = TFConv2d(c1, c_, 1, 1, bias=False, w=w.cv2)
221 self.cv3 = TFConv2d(c_, c_, 1, 1, bias=False, w=w.cv3)
222 self.cv4 = TFConv(2 * c_, c2, 1, 1, w=w.cv4)
223 self.bn = TFBN(w.bn)
224 self.act = lambda x: keras.activations.swish(x)
225 self.m = keras.Sequential([TFBottleneck(c_, c_, shortcut, g, e=1.0, w=w.m[j]) for j in range(n)])
226
227 def call(self, inputs):
228 y1 = self.cv3(self.m(self.cv1(inputs)))
229 y2 = self.cv2(inputs)
230 return self.cv4(self.act(self.bn(tf.concat((y1, y2), axis=3))))

295class TFDetect(keras.layers.Layer):
296 # TF YOLOv5 Detect layer
297 def __init__(self, nc=80, anchors=(), ch=(), imgsz=(640, 640), w=None): # detection layer
298 super().__init__()
299 self.stride = tf.convert_to_tensor(w.stride.numpy(), dtype=tf.float32)
300 self.nc = nc # number of classes
301 self.no = nc + 5 # number of outputs per anchor
302 self.nl = len(anchors) # number of detection layers
303 self.na = len(anchors[0]) // 2 # number of anchors
304 self.grid = [tf.zeros(1)] * self.nl # init grid
305 self.anchors = tf.convert_to_tensor(w.anchors.numpy(), dtype=tf.float32)
306 self.anchor_grid = tf.reshape(self.anchors * tf.reshape(self.stride, [self.nl, 1, 1]), [self.nl, 1, -1, 1, 2])
307 self.m = [TFConv2d(x, self.no * self.na, 1, w=w.m[i]) for i, x in enumerate(ch)]
308 self.training = False # set to False after building model
309 self.imgsz = imgsz
310 for i in range(self.nl):
311 ny, nx = self.imgsz[0] // self.stride[i], self.imgsz[1] // self.stride[i]
312 self.grid[i] = self._make_grid(nx, ny)
313
314 def call(self, inputs):
315 z = [] # inference output
316 x = []
317 for i in range(self.nl):
318 x.append(self.m[i](inputs[i]))
319 # x(bs,20,20,255) to x(bs,3,20,20,85)
320 ny, nx = self.imgsz[0] // self.stride[i], self.imgsz[1] // self.stride[i]
321 x[i] = tf.reshape(x[i], [-1, ny * nx, self.na, self.no])
322
323 if not self.training: # inference
324 y = x[i]
325 grid = tf.transpose(self.grid[i], [0, 2, 1, 3]) - 0.5
326 anchor_grid = tf.transpose(self.anchor_grid[i], [0, 2, 1, 3]) * 4
327 xy = (tf.sigmoid(y[..., 0:2]) * 2 + grid) * self.stride[i] # xy
328 wh = tf.sigmoid(y[..., 2:4]) ** 2 * anchor_grid
329 # Normalize xywh to 0-1 to reduce calibration error
330 xy /= tf.constant([[self.imgsz[1], self.imgsz[0]]], dtype=tf.float32)
331 wh /= tf.constant([[self.imgsz[1], self.imgsz[0]]], dtype=tf.float32)
332 y = tf.concat([xy, wh, tf.sigmoid(y[..., 4 : 5 + self.nc]), y[..., 5 + self.nc :]], -1)
333 z.append(tf.reshape(y, [-1, self.na * ny * nx, self.no]))
334
335 return tf.transpose(x, [0, 2, 1, 3]) if self.training else (tf.concat(z, 1),)
336
337 @staticmethod
338 def _make_grid(nx=20, ny=20):
339 # yv, xv = torch.meshgrid([torch.arange(ny), torch.arange(nx)])
340 # return torch.stack((xv, yv), 2).view((1, 1, ny, nx, 2)).float()
341 xv, yv = tf.meshgrid(tf.range(nx), tf.range(ny))
342 return tf.cast(tf.reshape(tf.stack([xv, yv], 2), [1, 1, ny * nx, 2]), dtype=tf.float32)

346 # YOLOv5 Segment head for segmentation models
347 def __init__(self, nc=80, anchors=(), nm=32, npr=256, ch=(), imgsz=(640, 640), w=None):
348 super().__init__(nc, anchors, ch, imgsz, w)

353 self.proto = TFProto(ch[0], self.npr, self.nm, w=w.proto) # protos

355

357 p = self.proto(x[0])
358 # p = TFUpsample(None, scale_factor=4, mode='nearest')(self.proto(x[0])) # (optional) full-size protos

402def parse_model(d, ch, model, imgsz): # model_dict, input_channels(3)
403 LOGGER.info(f"\n{'':>3}{'from':>18}{'n':>3}{'params':>10} {'module':<40}{'arguments':<30}")
404 anchors, nc, gd, gw, ch_mul = (
405 d["anchors"],
406 d["nc"],
407 d["depth_multiple"],
408 d["width_multiple"],
409 d.get("channel_multiple"),
410 )
411 na = (len(anchors[0]) // 2) if isinstance(anchors, list) else anchors # number of anchors
412 no = na * (nc + 5) # number of outputs = anchors * (classes + 5)
413 if not ch_mul:
414 ch_mul = 8
415
416 layers, save, c2 = [], [], ch[-1] # layers, savelist, ch out
417 for i, (f, n, m, args) in enumerate(d["backbone"] + d["head"]): # from, number, module, args
418 m_str = m
419 m = eval(m) if isinstance(m, str) else m # eval strings
420 for j, a in enumerate(args):
421 try:
422 args[j] = eval(a) if isinstance(a, str) else a # eval strings
423 except NameError:
424 pass
425
426 n = max(round(n * gd), 1) if n > 1 else n # depth gain
427 if m in [
428 nn.Conv2d,
429 Conv,
430 DWConv,
431 DWConvTranspose2d,
432 Bottleneck,
433 SPP,
434 SPPF,
435 MixConv2d,
436 Focus,
437 CrossConv,
438 BottleneckCSP,
439 C3,
440 C3x,
441 ]:
442 c1, c2 = ch[f], args[0]
443 c2 = make_divisible(c2 * gw, ch_mul) if c2 != no else c2
444
445 args = [c1, c2, *args[1:]]
446 if m in [BottleneckCSP, C3, C3x]:
447 args.insert(2, n)
448 n = 1
449 elif m is nn.BatchNorm2d:
450 args = [ch[f]]
451 elif m is Concat:
452 c2 = sum(ch[-1 if x == -1 else x + 1] for x in f)
453 elif m in [Detect, Segment]:
454 args.append([ch[x + 1] for x in f])
455 if isinstance(args[1], int): # number of anchors
456 args[1] = [list(range(args[1] * 2))] * len(f)
457 if m is Segment:
458 args[3] = make_divisible(args[3] * gw, ch_mul)
459 args.append(imgsz)
460 else:
461 c2 = ch[f]
462
463 tf_m = eval("TF" + m_str.replace("nn.", ""))
464 m_ = (
465 keras.Sequential([tf_m(*args, w=model.model[i][j]) for j in range(n)])
466 if n > 1
467 else tf_m(*args, w=model.model[i])
468 ) # module
469
470 torch_m_ = nn.Sequential(*(m(*args) for _ in range(n))) if n > 1 else m(*args) # module
471 t = str(m)[8:-2].replace("__main__.", "") # module type
472 np = sum(x.numel() for x in torch_m_.parameters()) # number params
473 m_.i, m_.f, m_.type, m_.np = i, f, t, np # attach index, 'from' index, type, number params
474 LOGGER.info(f"{i:>3}{str(f):>18}{str(n):>3}{np:>10} {t:<40}{str(args):<30}") # print
475 save.extend(x % i for x in ([f] if isinstance(f, int) else f) if x != -1) # append to savelist
476 layers.append(m_)
477 ch.append(c2)
478 return keras.Sequential(layers), sorted(save)

509 ):
510 y = [] # outputs
511 x = inputs
512 for m in self.model.layers:
513 if m.f != -1: # if not from previous layer
514 x = y[m.f] if isinstance(m.f, int) else [x if j == -1 else y[j] for j in m.f] # from earlier layers
515
516 x = m(x) # run
517 y.append(x if m.i in self.savelist else None) # save output
518
519 # Add TensorFlow NMS
520 if tf_nms:
521 boxes = self._xywh2xyxy(x[0][..., :4])
522 probs = x[0][:, :, 4:5]
523 classes = x[0][:, :, 5:]
524 scores = probs * classes
525 if agnostic_nms:
526 nms = AgnosticNMS()((boxes, classes, scores), topk_all, iou_thres, conf_thres)
527 else:
528 boxes = tf.expand_dims(boxes, 2)
529 nms = tf.image.combined_non_max_suppression(
530 boxes, scores, topk_per_class, topk_all, iou_thres, conf_thres, clip_boxes=False
531 )
532 return (nms,)
533 return x # output [1,6300,85] = [xywh, conf, class0, class1, ...]
534 # x = x[0] # [x(1,6300,85), ...] to x(6300,85)
535 # xywh = x[..., :4] # x(6300,4) boxes
536 # conf = x[..., 4:5] # x(6300,1) confidences
537 # cls = tf.reshape(tf.cast(tf.argmax(x[..., 5:], axis=1), tf.float32), (-1, 1)) # x(6300,1) classes
538 # return tf.concat([conf, cls, xywh], 1)
539
540 @staticmethod
541 def _xywh2xyxy(xywh):
542 # Convert nx4 boxes from [x, y, w, h] to [x1, y1, x2, y2] where xy1=top-left, xy2=bottom-right
543 x, y, w, h = tf.split(xywh, num_or_size_splits=4, axis=-1)
544 return tf.concat([x - w / 2, y - h / 2, x + w / 2, y + h / 2], axis=-1)

547class AgnosticNMS(keras.layers.Layer):
548 # TF Agnostic NMS
549 def call(self, input, topk_all, iou_thres, conf_thres):
550 # wrap map_fn to avoid TypeSpec related error https://stackoverflow.com/a/65809989/3036450
551 return tf.map_fn(
552 lambda x: self._nms(x, topk_all, iou_thres, conf_thres),
553 input,
554 fn_output_signature=(tf.float32, tf.float32, tf.float32, tf.int32),
555 name="agnostic_nms",
556 )
557
558 @staticmethod
559 def _nms(x, topk_all=100, iou_thres=0.45, conf_thres=0.25): # agnostic NMS
560 boxes, classes, scores = x
561 class_inds = tf.cast(tf.argmax(classes, axis=-1), tf.float32)
562 scores_inp = tf.reduce_max(scores, -1)
563 selected_inds = tf.image.non_max_suppression(
564 boxes, scores_inp, max_output_size=topk_all, iou_threshold=iou_thres, score_threshold=conf_thres
565 )
566 selected_boxes = tf.gather(boxes, selected_inds)
567 padded_boxes = tf.pad(
568 selected_boxes,
569 paddings=[[0, topk_all - tf.shape(selected_boxes)[0]], [0, 0]],
570 mode="CONSTANT",
571 constant_values=0.0,
572 )
573 selected_scores = tf.gather(scores_inp, selected_inds)
574 padded_scores = tf.pad(
575 selected_scores,
576 paddings=[[0, topk_all - tf.shape(selected_boxes)[0]]],
577 mode="CONSTANT",
578 constant_values=-1.0,
579 )
580 selected_classes = tf.gather(class_inds, selected_inds)
581 padded_classes = tf.pad(
582 selected_classes,
583 paddings=[[0, topk_all - tf.shape(selected_boxes)[0]]],
584 mode="CONSTANT",
585 constant_values=-1.0,
586 )
587 valid_detections = tf.shape(selected_inds)[0]
588 return padded_boxes, padded_scores, padded_classes, valid_detections

591def activations(act=nn.SiLU):
592 # Returns TF activation from input PyTorch activation
593 if isinstance(act, nn.LeakyReLU):
594 return lambda x: keras.activations.relu(x, alpha=0.1)
595 elif isinstance(act, nn.Hardswish):
596 return lambda x: x * tf.nn.relu6(x + 3) * 0.166666667
597 elif isinstance(act, (nn.SiLU, SiLU)):
598 return lambda x: keras.activations.swish(x)
599 else:
600 raise Exception(f"no matching TensorFlow activation found for PyTorch activation {act}")
