21def construct_dataset(clearml_info_string):
22 """Load in a clearml dataset and fill the internal data_dict with its contents."""
23 dataset_id = clearml_info_string.replace("clearml://", "")
24 dataset = Dataset.get(dataset_id=dataset_id)
25 dataset_root_path = Path(dataset.get_local_copy())
26
27 # We'll search for the yaml file definition in the dataset
28 yaml_filenames = list(glob.glob(str(dataset_root_path / "*.yaml")) + glob.glob(str(dataset_root_path / "*.yml")))
29 if len(yaml_filenames) > 1:
30 raise ValueError(
31 "More than one yaml file was found in the dataset root, cannot determine which one contains "
32 "the dataset definition this way."
33 )
34 elif not yaml_filenames:
35 raise ValueError(
36 "No yaml definition found in dataset root path, check that there is a correct yaml file "
37 "inside the dataset root path."
38 )
39 with open(yaml_filenames[0]) as f:
40 dataset_definition = yaml.safe_load(f)
41
42 assert set(
43 dataset_definition.keys()
44 ).issuperset(
45 {"train", "test", "val", "nc", "names"}
46 ), "The right keys were not found in the yaml file, make sure it at least has the following keys: ('train', 'test', 'val', 'nc', 'names')"
47
48 data_dict = {}
49 data_dict["train"] = (
50 str((dataset_root_path / dataset_definition["train"]).resolve()) if dataset_definition["train"] else None
51 )
52 data_dict["test"] = (
53 str((dataset_root_path / dataset_definition["test"]).resolve()) if dataset_definition["test"] else None
54 )
55 data_dict["val"] = (
56 str((dataset_root_path / dataset_definition["val"]).resolve()) if dataset_definition["val"] else None
57 )
58 data_dict["nc"] = dataset_definition["nc"]
59 data_dict["names"] = dataset_definition["names"]
60
61 return data_dict

64class ClearmlLogger:
65 """
66 Log training runs, datasets, models, and predictions to ClearML.
67
68 This logger sends information to ClearML at app.clear.ml or to your own hosted server. By default, this information
69 includes hyperparameters, system configuration and metrics, model metrics, code information and basic data metrics
70 and analyses.
71
72 By providing additional command line arguments to train.py, datasets, models and predictions can also be logged.
73 """

75 def __init__(self, opt, hyp):
76 """
77 - Initialize ClearML Task, this object will capture the experiment
78 - Upload dataset version to ClearML Data if opt.upload_dataset is True
79
80 arguments:
81 opt (namespace) -- Commandline arguments for this run
82 hyp (dict) -- Hyperparameters for this run
83
84 """
85 self.current_epoch = 0
86 # Keep tracked of amount of logged images to enforce a limit
87 self.current_epoch_logged_images = set()
88 # Maximum number of images to log to clearML per epoch
89 self.max_imgs_to_log_per_epoch = 16
90 # Get the interval of epochs when bounding box images should be logged
91 # Only for detection task though!
92 if "bbox_interval" in opt:
93 self.bbox_interval = opt.bbox_interval
94 self.clearml = clearml
95 self.task = None
96 self.data_dict = None
97 if self.clearml:
98 self.task = Task.init(
99 project_name="YOLOv5" if str(opt.project).startswith("runs/") else opt.project,
100 task_name=opt.name if opt.name != "exp" else "Training",
101 tags=["YOLOv5"],
102 output_uri=True,
103 reuse_last_task_id=opt.exist_ok,
104 auto_connect_frameworks={"pytorch": False, "matplotlib": False},
105 # We disconnect pytorch auto-detection, because we added manual model save points in the code
106 )
107 # ClearML's hooks will already grab all general parameters
108 # Only the hyperparameters coming from the yaml config file
109 # will have to be added manually!
110 self.task.connect(hyp, name="Hyperparameters")
111 self.task.connect(opt, name="Args")
112
113 # Make sure the code is easily remotely runnable by setting the docker image to use by the remote agent
114 self.task.set_base_docker(
115 "ultralytics/yolov5:latest",
116 docker_arguments='--ipc=host -e="CLEARML_AGENT_SKIP_PYTHON_ENV_INSTALL=1"',
117 docker_setup_bash_script="pip install clearml",
118 )
119
120 # Get ClearML Dataset Version if requested
121 if opt.data.startswith("clearml://"):
122 # data_dict should have the following keys:
123 # names, nc (number of classes), test, train, val (all three relative paths to ../datasets)
124 self.data_dict = construct_dataset(opt.data)
125 # Set data to data_dict because wandb will crash without this information and opt is the best way
126 # to give it to them
127 opt.data = self.data_dict

195 def log_image_with_boxes(self, image_path, boxes, class_names, image, conf_threshold=0.25):
196 """
197 Draw the bounding boxes on a single image and report the result as a ClearML debug sample.
198
199 arguments:
200 image_path (PosixPath) the path the original image file
201 boxes (list): list of scaled predictions in the format - [xmin, ymin, xmax, ymax, confidence, class]
202 class_names (dict): dict containing mapping of class int to class name
203 image (Tensor): A torch tensor containing the actual image data
204 """
205 if (
206 len(self.current_epoch_logged_images) < self.max_imgs_to_log_per_epoch
207 and self.current_epoch >= 0
208 and (self.current_epoch % self.bbox_interval == 0 and image_path not in self.current_epoch_logged_images)
209 ):
210 im = np.ascontiguousarray(np.moveaxis(image.mul(255).clamp(0, 255).byte().cpu().numpy(), 0, 2))
211 annotator = Annotator(im=im, pil=True)
212 for i, (conf, class_nr, box) in enumerate(zip(boxes[:, 4], boxes[:, 5], boxes[:, :4])):
213 color = colors(i)
214
215 class_name = class_names[int(class_nr)]
216 confidence_percentage = round(float(conf) * 100, 2)
217 label = f"{class_name}: {confidence_percentage}%"
218
219 if conf > conf_threshold:
220 annotator.rectangle(box.cpu().numpy(), outline=color)
221 annotator.box_label(box.cpu().numpy(), label=label, color=color)
222
223 annotated_image = annotator.result()
224 self.task.get_logger().report_image(
225 title="Bounding Boxes", series=image_path.name, iteration=self.current_epoch, image=annotated_image
226 )
227 self.current_epoch_logged_images.add(image_path)
