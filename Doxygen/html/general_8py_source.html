186class Profile(contextlib.ContextDecorator):
187 # YOLOv5 Profile class. Usage: @Profile() decorator or 'with Profile():' context manager
188 def __init__(self, t=0.0, device: torch.device = None):
189 self.t = t
190 self.device = device
191 self.cuda = bool(device and str(device).startswith("cuda"))

193 def __enter__(self):
194 self.start = self.time()
195 return self

198 self.dt = self.time() - self.start # delta-time
199 self.t += self.dt # accumulate dt

207class Timeout(contextlib.ContextDecorator):
208 # YOLOv5 Timeout class. Usage: @Timeout(seconds) decorator or 'with Timeout(seconds):' context manager
209 def __init__(self, seconds, *, timeout_msg="", suppress_timeout_errors=True):
210 self.seconds = int(seconds)
211 self.timeout_message = timeout_msg
212 self.suppress = bool(suppress_timeout_errors)
213
214 def _timeout_handler(self, signum, frame):
215 raise TimeoutError(self.timeout_message)
216
217 def __enter__(self):
218 if platform.system() != "Windows": # not supported on Windows
219 signal.signal(signal.SIGALRM, self._timeout_handler_timeout_handler) # Set handler for SIGALRM
220 signal.alarm(self.seconds) # start countdown for SIGALRM to be raised
221
222 def __exit__(self, exc_type, exc_val, exc_tb):
223 if platform.system() != "Windows":
224 signal.alarm(0) # Cancel SIGALRM if it's scheduled
225 if self.suppress and exc_type is TimeoutError: # Suppress TimeoutError
226 return True

560 def amp_allclose(model, im):
561 # All close FP32 vs AMP results
562 m = AutoShape(model, verbose=False) # model
563 a = m(im).xywhn[0] # FP32 inference
564 m.amp = True
565 b = m(im).xywhn[0] # AMP inference
566 return a.shape == b.shape and torch.allclose(a, b, atol=0.1) # close to 10% absolute tolerance
567
568 prefix = colorstr("AMP: ")
569 device = next(model.parameters()).device # get model device
570 if device.type in ("cpu", "mps"):
571 return False # AMP only used on CUDA devices
572 f = ROOT / "data" / "images" / "bus.jpg" # image to check
573 im = f if f.exists() else "https://ultralytics.com/images/bus.jpg" if check_online() else np.ones((640, 640, 3))

575 assert amp_allclose(deepcopy(model), im) or amp_allclose(DetectMultiBackend("yolov5n.pt", device), im)
576 LOGGER.info(f"{prefix}checks passed âœ…")
577 return True
578 except Exception:
579 help_url = "https://github.com/ultralytics/yolov5/issues/7908"
580 LOGGER.warning(f"{prefix}checks failed âŒ, disabling Automatic Mixed Precision. See {help_url}")
581 return False

612def download(url, dir=".", unzip=True, delete=True, curl=False, threads=1, retry=3):
613 # Multithreaded file download and unzip function, used in data.yaml for autodownload
614 def download_one(url, dir):
615 # Download 1 file
616 success = True
617 if os.path.isfile(url):
618 f = Path(url) # filename
619 else: # does not exist
620 f = dir / Path(url).name
621 LOGGER.info(f"Downloading {url} to {f}...")
622 for i in range(retry + 1):
623 if curl:
624 success = curl_download(url, f, silent=(threads > 1))
625 else:
626 torch.hub.download_url_to_file(url, f, progress=threads == 1) # torch download
627 success = f.is_file()
628 if success:
629 break
630 elif i < retry:
631 LOGGER.warning(f"âš ï¸ Download failure, retrying {i + 1}/{retry} {url}...")
632 else:
633 LOGGER.warning(f"âŒ Failed to download {url}...")
634
635 if unzip and success and (f.suffix == ".gz" or is_zipfile(f) or is_tarfile(f)):
636 LOGGER.info(f"Unzipping {f}...")
637 if is_zipfile(f):
638 unzip_file(f, dir) # unzip
639 elif is_tarfile(f):
640 subprocess.run(["tar", "xf", f, "--directory", f.parent], check=True) # unzip
641 elif f.suffix == ".gz":
642 subprocess.run(["tar", "xfz", f, "--directory", f.parent], check=True) # unzip
643 if delete:
644 f.unlink() # remove zip
645
646 dir = Path(dir)
647 dir.mkdir(parents=True, exist_ok=True) # make directory
648 if threads > 1:
649 pool = ThreadPool(threads)
650 pool.imap(lambda x: download_one(*x), zip(url, repeat(dir))) # multithreaded
651 pool.close()
652 pool.join()
653 else:
654 for u in [url] if isinstance(url, (str, Path)) else url:
655 download_one(u, dir)

818def xyxy2xywh(x):
819 # Convert nx4 boxes from [x1, y1, x2, y2] to [x, y, w, h] where xy1=top-left, xy2=bottom-right
820 y = x.clone() if isinstance(x, torch.Tensor) else np.copy(x)
821 y[..., 0] = (x[..., 0] + x[..., 2]) / 2 # x center
822 y[..., 1] = (x[..., 1] + x[..., 3]) / 2 # y center
823 y[..., 2] = x[..., 2] - x[..., 0] # width
824 y[..., 3] = x[..., 3] - x[..., 1] # height
825 return y

955def non_max_suppression(
956 prediction,
957 conf_thres=0.25,
958 iou_thres=0.45,
959 classes=None,
960 agnostic=False,
961 multi_label=False,
962 labels=(),
963 max_det=300,
964 nm=0, # number of masks
965):
966 """
967 Non-Maximum Suppression (NMS) on inference results to reject overlapping detections.
968
969 Returns:
970 list of detections, on (n,6) tensor per image [xyxy, conf, cls]
971 """
972
973 # Checks
974 assert 0 <= conf_thres <= 1, f"Invalid Confidence threshold {conf_thres}, valid values are between 0.0 and 1.0"
975 assert 0 <= iou_thres <= 1, f"Invalid IoU {iou_thres}, valid values are between 0.0 and 1.0"
976 if isinstance(prediction, (list, tuple)): # YOLOv5 model in validation model, output = (inference_out, loss_out)
977 prediction = prediction[0] # select only inference output
978
979 device = prediction.device
980 mps = "mps" in device.type # Apple MPS
981 if mps: # MPS not fully supported yet, convert tensors to CPU before NMS
982 prediction = prediction.cpu()
983 bs = prediction.shape[0] # batch size
984 nc = prediction.shape[2] - nm - 5 # number of classes
985 xc = prediction[..., 4] > conf_thres # candidates
986
987 # Settings
988 # min_wh = 2 # (pixels) minimum box width and height
989 max_wh = 7680 # (pixels) maximum box width and height
990 max_nms = 30000 # maximum number of boxes into torchvision.ops.nms()
991 time_limit = 0.5 + 0.05 * bs # seconds to quit after
992 redundant = True # require redundant detections
993 multi_label &= nc > 1 # multiple labels per box (adds 0.5ms/img)
994 merge = False # use merge-NMS
995
996 t = time.time()
997 mi = 5 + nc # mask start index
998 output = [torch.zeros((0, 6 + nm), device=prediction.device)] * bs
999 for xi, x in enumerate(prediction): # image index, image inference
1000 # Apply constraints
1001 # x[((x[..., 2:4] < min_wh) | (x[..., 2:4] > max_wh)).any(1), 4] = 0 # width-height
1002 x = x[xc[xi]] # confidence
1003
1004 # Cat apriori labels if autolabelling
1005 if labels and len(labels[xi]):
1006 lb = labels[xi]
1007 v = torch.zeros((len(lb), nc + nm + 5), device=x.device)
1008 v[:, :4] = lb[:, 1:5] # box
1009 v[:, 4] = 1.0 # conf
1010 v[range(len(lb)), lb[:, 0].long() + 5] = 1.0 # cls
1011 x = torch.cat((x, v), 0)
1012
1013 # If none remain process next image
1014 if not x.shape[0]:
1015 continue
1016
1017 # Compute conf
1018 x[:, 5:] *= x[:, 4:5] # conf = obj_conf * cls_conf
1019
1020 # Box/Mask
1021 box = xywh2xyxy(x[:, :4]) # center_x, center_y, width, height) to (x1, y1, x2, y2)
1022 mask = x[:, mi:] # zero columns if no masks
1023
1024 # Detections matrix nx6 (xyxy, conf, cls)
1025 if multi_label:
1026 i, j = (x[:, 5:mi] > conf_thres).nonzero(as_tuple=False).T
1027 x = torch.cat((box[i], x[i, 5 + j, None], j[:, None].float(), mask[i]), 1)
1028 else: # best class only
1029 conf, j = x[:, 5:mi].max(1, keepdim=True)
1030 x = torch.cat((box, conf, j.float(), mask), 1)[conf.view(-1) > conf_thres]
1031
1032 # Filter by class
1033 if classes is not None:
1034 x = x[(x[:, 5:6] == torch.tensor(classes, device=x.device)).any(1)]
1035
1036 # Apply finite constraint
1037 # if not torch.isfinite(x).all():
1038 # x = x[torch.isfinite(x).all(1)]
1039
1040 # Check shape
1041 n = x.shape[0] # number of boxes
1042 if not n: # no boxes
1043 continue
1044 x = x[x[:, 4].argsort(descending=True)[:max_nms]] # sort by confidence and remove excess boxes
1045
1046 # Batched NMS
1047 c = x[:, 5:6] * (0 if agnostic else max_wh) # classes
1048 boxes, scores = x[:, :4] + c, x[:, 4] # boxes (offset by class), scores
1049 i = torchvision.ops.nms(boxes, scores, iou_thres) # NMS
1050 i = i[:max_det] # limit detections
1051 if merge and (1 < n < 3e3): # Merge NMS (boxes merged using weighted mean)
1052 # update boxes as boxes(i,4) = weights(i,n) * boxes(n,4)
1053 iou = box_iou(boxes[i], boxes) > iou_thres # iou matrix
1054 weights = iou * scores[None] # box weights
1055 x[i, :4] = torch.mm(weights, x[:, :4]).float() / weights.sum(1, keepdim=True) # merged boxes
1056 if redundant:
1057 i = i[iou.sum(1) > 1] # require redundancy
1058
1059 output[xi] = x[i]
1060 if mps:
1061 output[xi] = output[xi].to(device)
1062 if (time.time() - t) > time_limit:
1063 LOGGER.warning(f"WARNING âš ï¸ NMS time limit {time_limit:.3f}s exceeded")
1064 break # time limit exceeded
1065
1066 return output
