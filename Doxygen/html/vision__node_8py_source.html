61# Function that applies color filtering on the image to return 2D points within the given block's bounding box
62def get_block_points(image, block):
63 # Create a mask to filter out colors in the image
64 mask = np.zeros(image.shape[:2],dtype=np.uint8)
65 image_hsv = cv.cvtColor(image, cv.COLOR_BGR2HSV)
66
67 # Apply the mask to the image
68 for color_range in color_ranges.values():
69 lower_color = np.array(color_range[0])
70 upper_color = np.array(color_range[1])
71 color_mask = cv.inRange(image_hsv, lower_color, upper_color)
72 mask = cv.bitwise_or(mask, color_mask)
73
74 # Apply the mask to the original image
75 result = cv.bitwise_and(image, image, mask=mask)
76 gray_image = cv.cvtColor(result, cv.COLOR_BGR2GRAY)
77 mask = cv.inRange(gray_image, 1, 255)
78
79 # Find non-black points in the mask
80 non_black_points = cv.findNonZero(mask)
81
82 block_points_2D = []
83
84 # If non-black points are found
85 if non_black_points is not None:
86 # Iterate through each non-black point
87 for point in non_black_points:
88 x, y = point[0]
89 # Adjust the coordinates to be relative to the block's position
90 block_points_2D.append([x + int(block[2]), y + int(block[3])])
91
92 return block_points_2D

105# Function that gets the coordinates of 3 vertices of the block's lowest surface
106def get_vertices(wf_points):
107 points=np.array(wf_points)
108
109 # Get only the block's lower surface points -> block's points close to the table's surface
110 z_value = table_height + tolerance_error
111 l_surface_points = points[abs(points[:,2] - z_value) <= block_error]
112
113 # Get the indexes of the points with the desired values (min y, max y, min x)
114 miny_index = np.argmin(l_surface_points[:,1])
115 maxy_index = np.argmax(l_surface_points[:,1])
116 minx_index = np.argmin(l_surface_points[:,0])
117
118 # Get the values of min y, max y and min x
119 miny_value = l_surface_points[miny_index, 1]
120 maxy_value = l_surface_points[maxy_index, 1]
121 minx_value = l_surface_points[minx_index, 0]
122
123 # Get only the points of the lower surface whose values are close to the one found above
124 miny_points = l_surface_points[abs(l_surface_points[:,1] - miny_value) <= value_error]
125 maxy_points = l_surface_points[abs(l_surface_points[:,1] - maxy_value) <= value_error]
126 minx_points = l_surface_points[abs(l_surface_points[:,0] - minx_value) <= value_error]
127
128 # Get the indexes of the points
129 miny_index_f = np.argmin(miny_points[:,0])
130 maxy_index_f = np.argmin(maxy_points[:,0])
131 minx_index_f = np.argmin(minx_points[:,1])
132
133 # Get the coordinates of the three points
134 miny_minx = miny_points[miny_index_f]
135 maxy_minx = maxy_points[maxy_index_f]

140# Function that calculates the angular value phi
141def get_block_pose(miny_minx, maxy_minx, minx):
142 # Coordinates of three vertices of the lower surface of the block
143 miny_minx = np.array(miny_minx)
144 maxy_minx = np.array(maxy_minx)
145 minx = np.array(minx)
146
147 miny_minx_dist = math.dist(miny_minx, minx)
148
149 if miny_minx_dist <= distance_error:
150 configuration = 1
151 else:
152 if maxy_minx[0] <= miny_minx[0]:
153 configuration = 2
154 else:
155 configuration = 3
156
157 if configuration == 2:
158 phi = abs(math.atan2(minx[1]-miny_minx[1],miny_minx[0]-minx[0]))
159 return phi
160 elif configuration == 3:
161 phi = abs(math.atan2(minx[1]-miny_minx[1],miny_minx[0]-minx[0]))
162 return phi - np.pi/2
163 elif configuration == 1:
164 return 0

166# Function that gets the coordinates of the central point of the block
167def get_block_center_point(min_x, phi, block_short_side, block_long_side, block_height):
168 x = min_x[0] + (block_long_side * math.cos(phi) + block_short_side * math.sin(abs(phi)))/2
169 y = min_x[1] - np.sign(phi) * (block_short_side * math.cos(phi) - block_long_side*math.sin(abs(phi)))/2
170 z = table_height + block_height/2
171
172 return x, y, z

178def object_detection(image_msg: Image, point_cloud2_msg: PointCloud2, model) -> None:
179 # convert received image (bgr8 format) to a cv2 image
180 img = CvBridge().imgmsg_to_cv2(image_msg, "bgr8")
181
182 # Apply model to the image
183 results = model(img)
184
185 # Get bounding box information of each block in the image
186 blocks = []
187 bboxes = results.pandas().xyxy[0].to_dict(orient="records")
188 for bbox in bboxes:
189 name = bbox['name']
190 block_short_side, block_long_side, block_height = get_block_values(name)
191
192 conf = bbox['confidence']
193 x1 = int(bbox['xmin'])
194 y1 = int(bbox['ymin'])
195 x2 = int(bbox['xmax'])
196 y2 = int(bbox['ymax'])
197
198 blocks.append((name, conf, x1, y1, x2, y2, block_short_side, block_long_side, block_height))
199
200 # Localize each detected block and get its pose with respect to the world frame
201 for block in blocks:
202 # Crop image -> take only the points inside block's bounding box
203 image = img[block[3]:block[5], block[2]:block[4]]
204
205 # Get 2D points of the block by filtering the image through a mask
206 block_points_2D = get_block_points(image, block)
207
208 # Convert from a list of tuples to a list of lists
209 zed_points = []
210 for point in list(block_points_2D):
211 zed_points.append([int(coordinate) for coordinate in point])
212
213 # Convert 2D points to 3D points using point cloud data
214 block_points_3D = convert_2D_to_3D(point_cloud2_msg, zed_points)
215
216 # Rotational matrix to go from the zed frame to the world frame
217 rot_matrix = np.array([[0., -0.49948, 0.86632],
218 [-1., 0., 0.],
219 [0., -0.86632, -0.49948]])
220
221 # Zed camera's position from the world frame
222 zed_pose = np.array([-0.4, 0.59, 1.4])
223
224 # Convert each block's point from camera frame to world frame
225 wf_block_points = []
226 for point in block_points_3D:
227 point = rot_matrix.dot(point) + zed_pose
228 point = np.array(point)
229 wf_block_points.append(point)
230
231 # Get block's data and store it
232 vertices_result = get_vertices(wf_block_points)
233 miny_minx, maxy_minx, minx = vertices_result
234 phi = get_block_pose(miny_minx, maxy_minx, minx)
235 x, y, z = get_block_center_point(minx, phi, block[6], block[7], block[8]) # block[6] = block_short_side, block[7] = block_long_side, block[8] = block_height
236 store_block_data(block[0], x, y, z, phi) # block[0] = block's class name
