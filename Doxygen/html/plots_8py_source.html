102def hist2d(x, y, n=100):
103 # 2d histogram used in labels.png and evolve.png
104 xedges, yedges = np.linspace(x.min(), x.max(), n), np.linspace(y.min(), y.max(), n)
105 hist, xedges, yedges = np.histogram2d(x, y, (xedges, yedges))
106 xidx = np.clip(np.digitize(x, xedges) - 1, 0, hist.shape[0] - 1)
107 yidx = np.clip(np.digitize(y, yedges) - 1, 0, hist.shape[1] - 1)
108 return np.log(hist[xidx, yidx])

111def butter_lowpass_filtfilt(data, cutoff=1500, fs=50000, order=5):
112 from scipy.signal import butter, filtfilt
113
114 # https://stackoverflow.com/questions/28536191/how-to-filter-smooth-with-scipy-numpy
115 def butter_lowpass(cutoff, fs, order):
116 nyq = 0.5 * fs
117 normal_cutoff = cutoff / nyq
118 return butter(order, normal_cutoff, btype="low", analog=False)
119
120 b, a = butter_lowpass(cutoff, fs, order=order)
121 return filtfilt(b, a, data) # forward-backward filter

124def output_to_target(output, max_det=300):
125 # Convert model output to target format [batch_id, class_id, x, y, w, h, conf] for plotting
126 targets = []
127 for i, o in enumerate(output):
128 box, conf, cls = o[:max_det, :6].cpu().split((4, 1, 1), 1)
129 j = torch.full((conf.shape[0], 1), i)
130 targets.append(torch.cat((j, cls, xyxy2xywh(box), conf), 1))
131 return torch.cat(targets, 0).numpy()

134@threaded
135def plot_images(images, targets, paths=None, fname="images.jpg", names=None):
136 # Plot image grid with labels
137 if isinstance(images, torch.Tensor):
138 images = images.cpu().float().numpy()
139 if isinstance(targets, torch.Tensor):
140 targets = targets.cpu().numpy()
141
142 max_size = 1920 # max image size
143 max_subplots = 16 # max image subplots, i.e. 4x4
144 bs, _, h, w = images.shape # batch size, _, height, width
145 bs = min(bs, max_subplots) # limit plot images
146 ns = np.ceil(bs**0.5) # number of subplots (square)
147 if np.max(images[0]) <= 1:
148 images *= 255 # de-normalise (optional)
149
150 # Build Image
151 mosaic = np.full((int(ns * h), int(ns * w), 3), 255, dtype=np.uint8) # init
152 for i, im in enumerate(images):
153 if i == max_subplots: # if last batch has fewer images than we expect
154 break
155 x, y = int(w * (i // ns)), int(h * (i % ns)) # block origin
156 im = im.transpose(1, 2, 0)
157 mosaic[y : y + h, x : x + w, :] = im
158
159 # Resize (optional)
160 scale = max_size / ns / max(h, w)
161 if scale < 1:
162 h = math.ceil(scale * h)
163 w = math.ceil(scale * w)
164 mosaic = cv2.resize(mosaic, tuple(int(x * ns) for x in (w, h)))
165
166 # Annotate
167 fs = int((h + w) * ns * 0.01) # font size
168 annotator = Annotator(mosaic, line_width=round(fs / 10), font_size=fs, pil=True, example=names)
169 for i in range(i + 1):
170 x, y = int(w * (i // ns)), int(h * (i % ns)) # block origin
171 annotator.rectangle([x, y, x + w, y + h], None, (255, 255, 255), width=2) # borders
172 if paths:
173 annotator.text([x + 5, y + 5], text=Path(paths[i]).name[:40], txt_color=(220, 220, 220)) # filenames
174 if len(targets) > 0:
175 ti = targets[targets[:, 0] == i] # image targets
176 boxes = xywh2xyxy(ti[:, 2:6]).T
177 classes = ti[:, 1].astype("int")
178 labels = ti.shape[1] == 6 # labels if no conf column
179 conf = None if labels else ti[:, 6] # check for confidence presence (label vs pred)
180
181 if boxes.shape[1]:
182 if boxes.max() <= 1.01: # if normalized with tolerance 0.01
183 boxes[[0, 2]] *= w # scale to pixels
184 boxes[[1, 3]] *= h
185 elif scale < 1: # absolute coords need scale if image scales
186 boxes *= scale
187 boxes[[0, 2]] += x
188 boxes[[1, 3]] += y
189 for j, box in enumerate(boxes.T.tolist()):
190 cls = classes[j]
191 color = colors(cls)
192 cls = names[cls] if names else cls
193 if labels or conf[j] > 0.25: # 0.25 conf thresh
194 label = f"{cls}" if labels else f"{cls} {conf[j]:.1f}"
195 annotator.box_label(box, label, color=color)
196 annotator.im.save(fname) # save

457def save_one_box(xyxy, im, file=Path("im.jpg"), gain=1.02, pad=10, square=False, BGR=False, save=True):
458 # Save image crop as {file} with crop size multiple {gain} and {pad} pixels. Save and/or return crop
459 xyxy = torch.tensor(xyxy).view(-1, 4)
460 b = xyxy2xywh(xyxy) # boxes
461 if square:
462 b[:, 2:] = b[:, 2:].max(1)[0].unsqueeze(1) # attempt rectangle to square
463 b[:, 2:] = b[:, 2:] * gain + pad # box wh * gain + pad
464 xyxy = xywh2xyxy(b).long()
465 clip_boxes(xyxy, im.shape)
466 crop = im[int(xyxy[0, 1]) : int(xyxy[0, 3]), int(xyxy[0, 0]) : int(xyxy[0, 2]), :: (1 if BGR else -1)]
467 if save:
468 file.parent.mkdir(parents=True, exist_ok=True) # make directory
469 f = str(increment_path(file).with_suffix(".jpg"))
470 # cv2.imwrite(f, crop) # save BGR, https://github.com/ultralytics/yolov5/issues/7007 chroma subsampling issue
471 Image.fromarray(crop[..., ::-1]).save(f, quality=95, subsampling=0) # save RGB
472 return crop
