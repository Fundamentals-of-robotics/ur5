12class Sum(nn.Module):
13 # Weighted sum of 2 or more layers https://arxiv.org/abs/1911.09070
14 def __init__(self, n, weight=False): # n: number of inputs
15 super().__init__()
16 self.weight = weight # apply weights boolean
17 self.iter = range(n - 1) # iter object
18 if weight:
19 self.w = nn.Parameter(-torch.arange(1.0, n) / 2, requires_grad=True) # layer weights
20
21 def forward(self, x):
22 y = x[0] # no weight
23 if self.weight:
24 w = torch.sigmoid(self.w) * 2
25 for i in self.iter:
26 y = y + x[i + 1] * w[i]
27 else:
28 for i in self.iter:
29 y = y + x[i + 1]
30 return y

33class MixConv2d(nn.Module):
34 # Mixed Depth-wise Conv https://arxiv.org/abs/1907.09595
35 def __init__(self, c1, c2, k=(1, 3), s=1, equal_ch=True): # ch_in, ch_out, kernel, stride, ch_strategy
36 super().__init__()
37 n = len(k) # number of convolutions
38 if equal_ch: # equal c_ per group
39 i = torch.linspace(0, n - 1e-6, c2).floor() # c2 indices
40 c_ = [(i == g).sum() for g in range(n)] # intermediate channels
41 else: # equal weight.numel() per group
42 b = [c2] + [0] * n
43 a = np.eye(n + 1, n, k=-1)
44 a -= np.roll(a, 1, axis=1)
45 a *= np.array(k) ** 2
46 a[0] = 1
47 c_ = np.linalg.lstsq(a, b, rcond=None)[0].round() # solve for equal weight indices, ax = b
48
49 self.m = nn.ModuleList(
50 [nn.Conv2d(c1, int(c_), k, s, k // 2, groups=math.gcd(c1, int(c_)), bias=False) for k, c_ in zip(k, c_)]
51 )
52 self.bn = nn.BatchNorm2d(c2)
53 self.act = nn.SiLU()
54
55 def forward(self, x):
56 return self.act(self.bn(torch.cat([m(x) for m in self.m], 1)))
