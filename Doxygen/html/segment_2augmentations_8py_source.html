14def mixup(im, labels, segments, im2, labels2, segments2):
15 # Applies MixUp augmentation https://arxiv.org/pdf/1710.09412.pdf
16 r = np.random.beta(32.0, 32.0) # mixup ratio, alpha=beta=32.0
17 im = (im * r + im2 * (1 - r)).astype(np.uint8)
18 labels = np.concatenate((labels, labels2), 0)
19 segments = np.concatenate((segments, segments2), 0)
20 return im, labels, segments

23def random_perspective(
24 im, targets=(), segments=(), degrees=10, translate=0.1, scale=0.1, shear=10, perspective=0.0, border=(0, 0)
25):
26 # torchvision.transforms.RandomAffine(degrees=(-10, 10), translate=(.1, .1), scale=(.9, 1.1), shear=(-10, 10))
27 # targets = [cls, xyxy]
28
29 height = im.shape[0] + border[0] * 2 # shape(h,w,c)
30 width = im.shape[1] + border[1] * 2
31
32 # Center
33 C = np.eye(3)
34 C[0, 2] = -im.shape[1] / 2 # x translation (pixels)
35 C[1, 2] = -im.shape[0] / 2 # y translation (pixels)
36
37 # Perspective
38 P = np.eye(3)
39 P[2, 0] = random.uniform(-perspective, perspective) # x perspective (about y)
40 P[2, 1] = random.uniform(-perspective, perspective) # y perspective (about x)
41
42 # Rotation and Scale
43 R = np.eye(3)
44 a = random.uniform(-degrees, degrees)
45 # a += random.choice([-180, -90, 0, 90]) # add 90deg rotations to small rotations
46 s = random.uniform(1 - scale, 1 + scale)
47 # s = 2 ** random.uniform(-scale, scale)
48 R[:2] = cv2.getRotationMatrix2D(angle=a, center=(0, 0), scale=s)
49
50 # Shear
51 S = np.eye(3)
52 S[0, 1] = math.tan(random.uniform(-shear, shear) * math.pi / 180) # x shear (deg)
53 S[1, 0] = math.tan(random.uniform(-shear, shear) * math.pi / 180) # y shear (deg)
54
55 # Translation
56 T = np.eye(3)
57 T[0, 2] = random.uniform(0.5 - translate, 0.5 + translate) * width # x translation (pixels)
58 T[1, 2] = random.uniform(0.5 - translate, 0.5 + translate) * height # y translation (pixels)
59
60 # Combined rotation matrix
61 M = T @ S @ R @ P @ C # order of operations (right to left) is IMPORTANT
62 if (border[0] != 0) or (border[1] != 0) or (M != np.eye(3)).any(): # image changed
63 if perspective:
64 im = cv2.warpPerspective(im, M, dsize=(width, height), borderValue=(114, 114, 114))
65 else: # affine
66 im = cv2.warpAffine(im, M[:2], dsize=(width, height), borderValue=(114, 114, 114))
67
68 # Visualize
69 # import matplotlib.pyplot as plt
70 # ax = plt.subplots(1, 2, figsize=(12, 6))[1].ravel()
71 # ax[0].imshow(im[:, :, ::-1]) # base
72 # ax[1].imshow(im2[:, :, ::-1]) # warped
73
74 # Transform label coordinates
75 n = len(targets)
76 new_segments = []
77 if n:
78 new = np.zeros((n, 4))
79 segments = resample_segments(segments) # upsample
80 for i, segment in enumerate(segments):
81 xy = np.ones((len(segment), 3))
82 xy[:, :2] = segment
83 xy = xy @ M.T # transform
84 xy = xy[:, :2] / xy[:, 2:3] if perspective else xy[:, :2] # perspective rescale or affine
85
86 # clip
87 new[i] = segment2box(xy, width, height)
88 new_segments.append(xy)
89
90 # filter candidates
91 i = box_candidates(box1=targets[:, 1:5].T * s, box2=new.T, area_thr=0.01)
92 targets = targets[i]
93 targets[:, 1:5] = new[i]
94 new_segments = np.array(new_segments)[i]
95
96 return im, targets, new_segments
