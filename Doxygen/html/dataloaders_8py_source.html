75def get_hash(paths):
76 # Returns a single hash value of a list of paths (files or dirs)
77 size = sum(os.path.getsize(p) for p in paths if os.path.exists(p)) # sizes
78 h = hashlib.sha256(str(size).encode()) # hash sizes
79 h.update("".join(paths).encode()) # hash paths
80 return h.hexdigest() # return hash

93def exif_transpose(image):
94 """
95 Transpose a PIL image accordingly if it has an EXIF Orientation tag.
96 Inplace version of https://github.com/python-pillow/Pillow/blob/master/src/PIL/ImageOps.py exif_transpose()
97
98 :param image: The image to transpose.
99 :return: An image.
100 """
101 exif = image.getexif()
102 orientation = exif.get(0x0112, 1) # default 1
103 if orientation > 1:
104 method = {
105 2: Image.FLIP_LEFT_RIGHT,
106 3: Image.ROTATE_180,
107 4: Image.FLIP_TOP_BOTTOM,
108 5: Image.TRANSPOSE,
109 6: Image.ROTATE_270,
110 7: Image.TRANSVERSE,
111 8: Image.ROTATE_90,
112 }.get(orientation)
113 if method is not None:
114 image = image.transpose(method)
115 del exif[0x0112]
116 image.info["exif"] = exif.tobytes()
117 return image

128# https://github.com/pytorch/pytorch/blob/master/torch/utils/data/distributed.py
129class SmartDistributedSampler(distributed.DistributedSampler):
130 def __iter__(self):
131 # deterministically shuffle based on epoch and seed
132 g = torch.Generator()
133 g.manual_seed(self.seed + self.epoch)
134
135 # determine the the eventual size (n) of self.indices (DDP indices)
136 n = int((len(self.dataset) - self.rank - 1) / self.num_replicas) + 1 # num_replicas == WORLD_SIZE
137 idx = torch.randperm(n, generator=g)
138 if not self.shuffle:
139 idx = idx.sort()[0]
140
141 idx = idx.tolist()
142 if self.drop_last:
143 idx = idx[: self.num_samples]
144 else:
145 padding_size = self.num_samples - len(idx)
146 if padding_size <= len(idx):
147 idx += idx[:padding_size]
148 else:
149 idx += (idx * math.ceil(padding_size / len(idx)))[:padding_size]
150
151 return iter(idx)

403class LoadStreams:
404 # YOLOv5 streamloader, i.e. `python detect.py --source 'rtsp://example.com/media.mp4' # RTSP, RTMP, HTTP streams`
405 def __init__(self, sources="file.streams", img_size=640, stride=32, auto=True, transforms=None, vid_stride=1):
406 torch.backends.cudnn.benchmark = True # faster for fixed-size inference
407 self.mode = "stream"
408 self.img_size = img_size
409 self.stride = stride
410 self.vid_stride = vid_stride # video frame-rate stride
411 sources = Path(sources).read_text().rsplit() if os.path.isfile(sources) else [sources]
412 n = len(sources)
413 self.sources = [clean_str(x) for x in sources] # clean source names for later
414 self.imgs, self.fps, self.frames, self.threads = [None] * n, [0] * n, [0] * n, [None] * n
415 for i, s in enumerate(sources): # index, source
416 # Start thread to read frames from video stream
417 st = f"{i + 1}/{n}: {s}... "
418 if urlparse(s).hostname in ("www.youtube.com", "youtube.com", "youtu.be"): # if source is YouTube video
419 # YouTube format i.e. 'https://www.youtube.com/watch?v=Zgi9g1ksQHc' or 'https://youtu.be/LNwODJXcvt4'
420 check_requirements(("pafy", "youtube_dl==2020.12.2"))
421 import pafy
422
423 s = pafy.new(s).getbest(preftype="mp4").url # YouTube URL
424 s = eval(s) if s.isnumeric() else s # i.e. s = '0' local webcam
425 if s == 0:
426 assert not is_colab(), "--source 0 webcam unsupported on Colab. Rerun command in a local environment."
427 assert not is_kaggle(), "--source 0 webcam unsupported on Kaggle. Rerun command in a local environment."
428 cap = cv2.VideoCapture(s)
429 assert cap.isOpened(), f"{st}Failed to open {s}"
430 w = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
431 h = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
432 fps = cap.get(cv2.CAP_PROP_FPS) # warning: may return 0 or nan
433 self.frames[i] = max(int(cap.get(cv2.CAP_PROP_FRAME_COUNT)), 0) or float("inf") # infinite stream fallback
434 self.fps[i] = max((fps if math.isfinite(fps) else 0) % 100, 0) or 30 # 30 FPS fallback
435
436 _, self.imgs[i] = cap.read() # guarantee first frame
437 self.threads[i] = Thread(target=self.update, args=([i, cap, s]), daemon=True)
438 LOGGER.info(f"{st} Success ({self.frames[i]} frames {w}x{h} at {self.fps[i]:.2f} FPS)")
439 self.threads[i].start()
440 LOGGER.info("") # newline
441
442 # check for common shapes
443 s = np.stack([letterbox(x, img_size, stride=stride, auto=auto)[0].shape for x in self.imgs])
444 self.rect = np.unique(s, axis=0).shape[0] == 1 # rect inference if all shapes equal
445 self.auto = auto and self.rect
446 self.transforms = transforms # optional
447 if not self.rect:
448 LOGGER.warning("WARNING ‚ö†Ô∏è Stream shapes differ. For optimal performance supply similarly-shaped streams.")
449
450 def update(self, i, cap, stream):
451 # Read stream `i` frames in daemon thread
452 n, f = 0, self.frames[i] # frame number, frame array
453 while cap.isOpened() and n < f:
454 n += 1
455 cap.grab() # .read() = .grab() followed by .retrieve()
456 if n % self.vid_stride == 0:
457 success, im = cap.retrieve()
458 if success:
459 self.imgs[i] = im
460 else:
461 LOGGER.warning("WARNING ‚ö†Ô∏è Video stream unresponsive, please check your IP camera connection.")
462 self.imgs[i] = np.zeros_like(self.imgs[i])
463 cap.open(stream) # re-open stream if signal was lost
464 time.sleep(0.0) # wait time

466 def __iter__(self):
467 self.count = -1
468 return self

470 def __next__(self):
471 self.count += 1
472 if not all(x.is_alive() for x in self.threads) or cv2.waitKey(1) == ord("q"): # q to quit
473 cv2.destroyAllWindows()
474 raise StopIteration
475
476 im0 = self.imgs.copy()
477 if self.transforms:
478 im = np.stack([self.transforms(x) for x in im0]) # transforms
479 else:
480 im = np.stack([letterbox(x, self.img_size, stride=self.stride, auto=self.auto)[0] for x in im0]) # resize
481 im = im[..., ::-1].transpose((0, 3, 1, 2)) # BGR to RGB, BHWC to BCHW
482 im = np.ascontiguousarray(im) # contiguous
483
484 return self.sources, im, im0, None, ""
