21def smooth(y, f=0.05):
22 # Box filter of fraction f
23 nf = round(len(y) * f * 2) // 2 + 1 # number of filter elements (must be odd)
24 p = np.ones(nf // 2) # ones padding
25 yp = np.concatenate((p * y[0], y, p * y[-1]), 0) # y padded
26 return np.convolve(yp, np.ones(nf) / nf, mode="valid") # y-smoothed

29def ap_per_class(tp, conf, pred_cls, target_cls, plot=False, save_dir=".", names=(), eps=1e-16, prefix=""):
30 """
31 Compute the average precision, given the recall and precision curves.
32
33 Source: https://github.com/rafaelpadilla/Object-Detection-Metrics.
34 # Arguments
35 tp: True positives (nparray, nx1 or nx10).
36 conf: Objectness value from 0-1 (nparray).
37 pred_cls: Predicted object classes (nparray).
38 target_cls: True object classes (nparray).
39 plot: Plot precision-recall curve at mAP@0.5
40 save_dir: Plot save directory
41 # Returns
42 The average precision as computed in py-faster-rcnn.
43 """
44
45 # Sort by objectness
46 i = np.argsort(-conf)
47 tp, conf, pred_cls = tp[i], conf[i], pred_cls[i]
48
49 # Find unique classes
50 unique_classes, nt = np.unique(target_cls, return_counts=True)
51 nc = unique_classes.shape[0] # number of classes, number of detections
52
53 # Create Precision-Recall curve and compute AP for each class
54 px, py = np.linspace(0, 1, 1000), [] # for plotting
55 ap, p, r = np.zeros((nc, tp.shape[1])), np.zeros((nc, 1000)), np.zeros((nc, 1000))
56 for ci, c in enumerate(unique_classes):
57 i = pred_cls == c
58 n_l = nt[ci] # number of labels
59 n_p = i.sum() # number of predictions
60 if n_p == 0 or n_l == 0:
61 continue
62
63 # Accumulate FPs and TPs
64 fpc = (1 - tp[i]).cumsum(0)
65 tpc = tp[i].cumsum(0)
66
67 # Recall
68 recall = tpc / (n_l + eps) # recall curve
69 r[ci] = np.interp(-px, -conf[i], recall[:, 0], left=0) # negative x, xp because xp decreases
70
71 # Precision
72 precision = tpc / (tpc + fpc) # precision curve
73 p[ci] = np.interp(-px, -conf[i], precision[:, 0], left=1) # p at pr_score
74
75 # AP from recall-precision curve
76 for j in range(tp.shape[1]):
77 ap[ci, j], mpre, mrec = compute_ap(recall[:, j], precision[:, j])
78 if plot and j == 0:
79 py.append(np.interp(px, mrec, mpre)) # precision at mAP@0.5
80
81 # Compute F1 (harmonic mean of precision and recall)
82 f1 = 2 * p * r / (p + r + eps)
83 names = [v for k, v in names.items() if k in unique_classes] # list: only classes that have data
84 names = dict(enumerate(names)) # to dict
85 if plot:
86 plot_pr_curve(px, py, ap, Path(save_dir) / f"{prefix}PR_curve.png", names)
87 plot_mc_curve(px, f1, Path(save_dir) / f"{prefix}F1_curve.png", names, ylabel="F1")
88 plot_mc_curve(px, p, Path(save_dir) / f"{prefix}P_curve.png", names, ylabel="Precision")
89 plot_mc_curve(px, r, Path(save_dir) / f"{prefix}R_curve.png", names, ylabel="Recall")
90
91 i = smooth(f1.mean(0), 0.1).argmax() # max F1 index
92 p, r, f1 = p[:, i], r[:, i], f1[:, i]
93 tp = (r * nt).round() # true positives
94 fp = (tp / (p + eps) - tp).round() # false positives
95 return tp, fp, p, r, f1, ap, unique_classes.astype(int)

98def compute_ap(recall, precision):
99 """Compute the average precision, given the recall and precision curves
100 # Arguments
101 recall: The recall curve (list)
102 precision: The precision curve (list)
103 # Returns
104 Average precision, precision curve, recall curve
105 """
106
107 # Append sentinel values to beginning and end
108 mrec = np.concatenate(([0.0], recall, [1.0]))
109 mpre = np.concatenate(([1.0], precision, [0.0]))
110
111 # Compute the precision envelope
112 mpre = np.flip(np.maximum.accumulate(np.flip(mpre)))
113
114 # Integrate area under curve
115 method = "interp" # methods: 'continuous', 'interp'
116 if method == "interp":
117 x = np.linspace(0, 1, 101) # 101-point interp (COCO)
118 ap = np.trapz(np.interp(x, mrec, mpre), x) # integrate
119 else: # 'continuous'
120 i = np.where(mrec[1:] != mrec[:-1])[0] # points where x axis (recall) changes
121 ap = np.sum((mrec[i + 1] - mrec[i]) * mpre[i + 1]) # area under curve
122
123 return ap, mpre, mrec

128 def __init__(self, nc, conf=0.25, iou_thres=0.45):
129 self.matrix = np.zeros((nc + 1, nc + 1))
130 self.nc = nc # number of classes
131 self.conf = conf
132 self.iou_thres = iou_thres

134 def process_batch(self, detections, labels):
135 """
136 Return intersection-over-union (Jaccard index) of boxes.
137
138 Both sets of boxes are expected to be in (x1, y1, x2, y2) format.
139 Arguments:
140 detections (Array[N, 6]), x1, y1, x2, y2, conf, class
141 labels (Array[M, 5]), class, x1, y1, x2, y2
142 Returns:
143 None, updates confusion matrix accordingly
144 """
145 if detections is None:
146 gt_classes = labels.int()
147 for gc in gt_classes:
148 self.matrix[self.nc, gc] += 1 # background FN
149 return
150
151 detections = detections[detections[:, 4] > self.conf]
152 gt_classes = labels[:, 0].int()
153 detection_classes = detections[:, 5].int()
154 iou = box_iou(labels[:, 1:], detections[:, :4])
155
156 x = torch.where(iou > self.iou_thres)
157 if x[0].shape[0]:
158 matches = torch.cat((torch.stack(x, 1), iou[x[0], x[1]][:, None]), 1).cpu().numpy()
159 if x[0].shape[0] > 1:
160 matches = matches[matches[:, 2].argsort()[::-1]]
161 matches = matches[np.unique(matches[:, 1], return_index=True)[1]]
162 matches = matches[matches[:, 2].argsort()[::-1]]
163 matches = matches[np.unique(matches[:, 0], return_index=True)[1]]
164 else:
165 matches = np.zeros((0, 3))
166
167 n = matches.shape[0] > 0
168 m0, m1, _ = matches.transpose().astype(int)
169 for i, gc in enumerate(gt_classes):
170 j = m0 == i
171 if n and sum(j) == 1:
172 self.matrix[detection_classes[m1[j]], gc] += 1 # correct
173 else:
174 self.matrix[self.nc, gc] += 1 # true background
175
176 if n:
177 for i, dc in enumerate(detection_classes):
178 if not any(m1 == i):
179 self.matrix[dc, self.nc] += 1 # predicted background

181 def tp_fp(self):
182 tp = self.matrix.diagonal() # true positives
183 fp = self.matrix.sum(1) - tp # false positives
184 # fn = self.matrix.sum(0) - tp # false negatives (missed detections)
185 return tp[:-1], fp[:-1] # remove background class

224def bbox_iou(box1, box2, xywh=True, GIoU=False, DIoU=False, CIoU=False, eps=1e-7):
225 # Returns Intersection over Union (IoU) of box1(1,4) to box2(n,4)
226
227 # Get the coordinates of bounding boxes
228 if xywh: # transform from xywh to xyxy
229 (x1, y1, w1, h1), (x2, y2, w2, h2) = box1.chunk(4, -1), box2.chunk(4, -1)
230 w1_, h1_, w2_, h2_ = w1 / 2, h1 / 2, w2 / 2, h2 / 2
231 b1_x1, b1_x2, b1_y1, b1_y2 = x1 - w1_, x1 + w1_, y1 - h1_, y1 + h1_
232 b2_x1, b2_x2, b2_y1, b2_y2 = x2 - w2_, x2 + w2_, y2 - h2_, y2 + h2_
233 else: # x1, y1, x2, y2 = box1
234 b1_x1, b1_y1, b1_x2, b1_y2 = box1.chunk(4, -1)
235 b2_x1, b2_y1, b2_x2, b2_y2 = box2.chunk(4, -1)
236 w1, h1 = b1_x2 - b1_x1, (b1_y2 - b1_y1).clamp(eps)
237 w2, h2 = b2_x2 - b2_x1, (b2_y2 - b2_y1).clamp(eps)
238
239 # Intersection area
240 inter = (b1_x2.minimum(b2_x2) - b1_x1.maximum(b2_x1)).clamp(0) * (
241 b1_y2.minimum(b2_y2) - b1_y1.maximum(b2_y1)
242 ).clamp(0)
243
244 # Union Area
245 union = w1 * h1 + w2 * h2 - inter + eps
246
247 # IoU
248 iou = inter / union
249 if CIoU or DIoU or GIoU:
250 cw = b1_x2.maximum(b2_x2) - b1_x1.minimum(b2_x1) # convex (smallest enclosing box) width
251 ch = b1_y2.maximum(b2_y2) - b1_y1.minimum(b2_y1) # convex height
252 if CIoU or DIoU: # Distance or Complete IoU https://arxiv.org/abs/1911.08287v1
253 c2 = cw**2 + ch**2 + eps # convex diagonal squared
254 rho2 = ((b2_x1 + b2_x2 - b1_x1 - b1_x2) ** 2 + (b2_y1 + b2_y2 - b1_y1 - b1_y2) ** 2) / 4 # center dist ** 2
255 if CIoU: # https://github.com/Zzh-tju/DIoU-SSD-pytorch/blob/master/utils/box/box_utils.py#L47
256 v = (4 / math.pi**2) * (torch.atan(w2 / h2) - torch.atan(w1 / h1)).pow(2)
257 with torch.no_grad():
258 alpha = v / (v - iou + (1 + eps))
259 return iou - (rho2 / c2 + v * alpha) # CIoU
260 return iou - rho2 / c2 # DIoU
261 c_area = cw * ch + eps # convex area
262 return iou - (c_area - union) / c_area # GIoU https://arxiv.org/pdf/1902.09630.pdf
263 return iou # IoU

266def box_iou(box1, box2, eps=1e-7):
267 # https://github.com/pytorch/vision/blob/master/torchvision/ops/boxes.py
268 """
269 Return intersection-over-union (Jaccard index) of boxes.
270
271 Both sets of boxes are expected to be in (x1, y1, x2, y2) format.
272 Arguments:
273 box1 (Tensor[N, 4])
274 box2 (Tensor[M, 4])
275 Returns:
276 iou (Tensor[N, M]): the NxM matrix containing the pairwise
277 IoU values for every element in boxes1 and boxes2
278 """
279
280 # inter(N,M) = (rb(N,M,2) - lt(N,M,2)).clamp(0).prod(2)
281 (a1, a2), (b1, b2) = box1.unsqueeze(1).chunk(2, 2), box2.unsqueeze(0).chunk(2, 2)
282 inter = (torch.min(a2, b2) - torch.max(a1, b1)).clamp(0).prod(2)
283
284 # IoU = inter / (area1 + area2 - inter)
285 return inter / ((a2 - a1).prod(2) + (b2 - b1).prod(2) - inter + eps)

288def bbox_ioa(box1, box2, eps=1e-7):
289 """
290 Returns the intersection over box2 area given box1, box2.
291
292 Boxes are x1y1x2y2
293 box1: np.array of shape(4)
294 box2: np.array of shape(nx4)
295 returns: np.array of shape(n)
296 """
297
298 # Get the coordinates of bounding boxes
299 b1_x1, b1_y1, b1_x2, b1_y2 = box1
300 b2_x1, b2_y1, b2_x2, b2_y2 = box2.T
301
302 # Intersection area
303 inter_area = (np.minimum(b1_x2, b2_x2) - np.maximum(b1_x1, b2_x1)).clip(0) * (
304 np.minimum(b1_y2, b2_y2) - np.maximum(b1_y1, b2_y1)
305 ).clip(0)
306
307 # box2 area
308 box2_area = (b2_x2 - b2_x1) * (b2_y2 - b2_y1) + eps
309
310 # Intersection over box2 area
311 return inter_area / box2_area

314def wh_iou(wh1, wh2, eps=1e-7):
315 # Returns the nxm IoU matrix. wh1 is nx2, wh2 is mx2
316 wh1 = wh1[:, None] # [N,1,2]
317 wh2 = wh2[None] # [1,M,2]
318 inter = torch.min(wh1, wh2).prod(2) # [N,M]
319 return inter / (wh1.prod(2) + wh2.prod(2) - inter + eps) # iou = inter / (area1 + area2 - inter)
