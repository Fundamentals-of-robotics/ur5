249 def log_predictions(self, image, labelsn, path, shape, predn):
250 if self.logged_images_count >= self.max_images:
251 return
252 detections = predn[predn[:, 4] > self.conf_thres]
253 iou = box_iou(labelsn[:, 1:], detections[:, :4])
254 mask, _ = torch.where(iou > self.iou_thres)
255 if len(mask) == 0:
256 return
257
258 filtered_detections = detections[mask]
259 filtered_labels = labelsn[mask]
260
261 image_id = path.split("/")[-1].split(".")[0]
262 image_name = f"{image_id}_curr_epoch_{self.experiment.curr_epoch}"
263 if image_name not in self.logged_image_names:
264 native_scale_image = PIL.Image.open(path)
265 self.log_image(native_scale_image, name=image_name)
266 self.logged_image_names.append(image_name)
267
268 metadata = [
269 {
270 "label": f"{self.class_names[int(cls)]}-gt",
271 "score": 100,
272 "box": {"x": xyxy[0], "y": xyxy[1], "x2": xyxy[2], "y2": xyxy[3]},
273 }
274 for cls, *xyxy in filtered_labels.tolist()
275 ]
276 metadata.extend(
277 {

289 def preprocess_prediction(self, image, labels, shape, pred):
290 nl, _ = labels.shape[0], pred.shape[0]
291
292 # Predictions
293 if self.opt.single_cls:
294 pred[:, 5] = 0
295
296 predn = pred.clone()
297 scale_boxes(image.shape[1:], predn[:, :4], shape[0], shape[1])
298
299 labelsn = None
300 if nl:
301 tbox = xywh2xyxy(labels[:, 1:5]) # target boxes
302 scale_boxes(image.shape[1:], tbox, shape[0], shape[1]) # native-space labels
303 labelsn = torch.cat((labels[:, 0:1], tbox), 1) # native-space labels
304 scale_boxes(image.shape[1:], predn[:, :4], shape[0], shape[1]) # native-space pred
305
306 return predn, labelsn
