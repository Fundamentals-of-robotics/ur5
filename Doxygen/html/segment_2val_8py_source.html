73def save_one_txt(predn, save_conf, shape, file):
74 # Save one txt result
75 gn = torch.tensor(shape)[[1, 0, 1, 0]] # normalization gain whwh
76 for *xyxy, conf, cls in predn.tolist():
77 xywh = (xyxy2xywh(torch.tensor(xyxy).view(1, 4)) / gn).view(-1).tolist() # normalized xywh
78 line = (cls, *xywh, conf) if save_conf else (cls, *xywh) # label format
79 with open(file, "a") as f:
80 f.write(("%g " * len(line)).rstrip() % line + "\n")

83def save_one_json(predn, jdict, path, class_map, pred_masks):
84 # Save one JSON result {"image_id": 42, "category_id": 18, "bbox": [258.15, 41.29, 348.26, 243.78], "score": 0.236}
85 from pycocotools.mask import encode
86
87 def single_encode(x):
88 rle = encode(np.asarray(x[:, :, None], order="F", dtype="uint8"))[0]
89 rle["counts"] = rle["counts"].decode("utf-8")
90 return rle
91
92 image_id = int(path.stem) if path.stem.isnumeric() else path.stem
93 box = xyxy2xywh(predn[:, :4]) # xywh
94 box[:, :2] -= box[:, 2:] / 2 # xy center to top-left corner
95 pred_masks = np.transpose(pred_masks, (2, 0, 1))
96 with ThreadPool(NUM_THREADS) as pool:
97 rles = pool.map(single_encode, pred_masks)
98 for i, (p, b) in enumerate(zip(predn.tolist(), box.tolist())):
99 jdict.append(
100 {
101 "image_id": image_id,
102 "category_id": class_map[int(p[5])],
103 "bbox": [round(x, 3) for x in b],
104 "score": round(p[4], 5),
105 "segmentation": rles[i],
106 }
107 )

110def process_batch(detections, labels, iouv, pred_masks=None, gt_masks=None, overlap=False, masks=False):
111 """
112 Return correct prediction matrix
113 Arguments:
114 detections (array[N, 6]), x1, y1, x2, y2, conf, class
115 labels (array[M, 5]), class, x1, y1, x2, y2
116 Returns:
117 correct (array[N, 10]), for 10 IoU levels
118 """
119 if masks:
120 if overlap:
121 nl = len(labels)
122 index = torch.arange(nl, device=gt_masks.device).view(nl, 1, 1) + 1
123 gt_masks = gt_masks.repeat(nl, 1, 1) # shape(1,640,640) -> (n,640,640)
124 gt_masks = torch.where(gt_masks == index, 1.0, 0.0)
125 if gt_masks.shape[1:] != pred_masks.shape[1:]:
126 gt_masks = F.interpolate(gt_masks[None], pred_masks.shape[1:], mode="bilinear", align_corners=False)[0]
127 gt_masks = gt_masks.gt_(0.5)
128 iou = mask_iou(gt_masks.view(gt_masks.shape[0], -1), pred_masks.view(pred_masks.shape[0], -1))
129 else: # boxes
130 iou = box_iou(labels[:, 1:], detections[:, :4])
131
132 correct = np.zeros((detections.shape[0], iouv.shape[0])).astype(bool)
133 correct_class = labels[:, 0:1] == detections[:, 5]
134 for i in range(len(iouv)):
135 x = torch.where((iou >= iouv[i]) & correct_class) # IoU > threshold and classes match
136 if x[0].shape[0]:
137 matches = torch.cat((torch.stack(x, 1), iou[x[0], x[1]][:, None]), 1).cpu().numpy() # [label, detect, iou]
138 if x[0].shape[0] > 1:
139 matches = matches[matches[:, 2].argsort()[::-1]]
140 matches = matches[np.unique(matches[:, 1], return_index=True)[1]]
141 # matches = matches[matches[:, 2].argsort()[::-1]]
142 matches = matches[np.unique(matches[:, 0], return_index=True)[1]]
143 correct[matches[:, 1].astype(int), i] = True
144 return torch.tensor(correct, dtype=torch.bool, device=iouv.device)

154 iou_thres=0.6, # NMS IoU threshold
155 max_det=300, # maximum detections per image
156 task="val", # train, val, test, speed or study
157 device="", # cuda device, i.e. 0 or 0,1,2,3 or cpu
158 workers=8, # max dataloader workers (per RANK in DDP mode)
159 single_cls=False, # treat as single-class dataset
160 augment=False, # augmented inference
161 verbose=False, # verbose output
162 save_txt=False, # save results to *.txt
163 save_hybrid=False, # save label+prediction hybrid results to *.txt
164 save_conf=False, # save confidences in --save-txt labels

190 half &= device.type != "cpu" # half precision only supported on CUDA
191 model.half() if half else model.float()
192 nm = de_parallel(model).model[-1].nm # number of masks

201 model = DetectMultiBackend(weights, device=device, dnn=dnn, data=data, fp16=half)
202 stride, pt, jit, engine = model.stride, model.pt, model.jit, model.engine
203 imgsz = check_img_size(imgsz, s=stride) # check image size
204 half = model.fp16 # FP16 supported on limited backends with CUDA
205 nm = de_parallel(model).model.model[-1].nm if isinstance(model, SegmentationModel) else 32 # number of masks
206 if engine:
207 batch_size = model.batch_size
208 else:
209 device = model.device
210 if not (pt or jit):
211 batch_size = 1 # export.py models default to batch-size 1
212 LOGGER.info(f"Forcing --batch-size 1 square inference (1,3,{imgsz},{imgsz}) for non-PyTorch models")
213
214 # Data
215 data = check_dataset(data) # check
216
217 # Configure
218 model.eval()
219 cuda = device.type != "cpu"
220 is_coco = isinstance(data.get("val"), str) and data["val"].endswith(f"coco{os.sep}val2017.txt") # COCO dataset
221 nc = 1 if single_cls else int(data["nc"]) # number of classes
222 iouv = torch.linspace(0.5, 0.95, 10, device=device) # iou vector for mAP@0.5:0.95
223 niou = iouv.numel()
224
225 # Dataloader
226 if not training:
227 if pt and not single_cls: # check --weights are trained on --data
228 ncm = model.model.nc
229 assert ncm == nc, (
230 f"{weights} ({ncm} classes) trained on different --data than what you passed ({nc} "
231 f"classes). Pass correct combination of --weights and --data that are trained together."
232 )
233 model.warmup(imgsz=(1 if pt else batch_size, 3, imgsz, imgsz)) # warmup
234 pad, rect = (0.0, False) if task == "speed" else (0.5, pt) # square inference for benchmarks
235 task = task if task in ("train", "val", "test") else "val" # path to train/val/test images
236 dataloader = create_dataloader(
237 data[task],
238 imgsz,
239 batch_size,
240 stride,
241 single_cls,
242 pad=pad,
243 rect=rect,
244 workers=workers,
245 prefix=colorstr(f"{task}: "),
246 overlap_mask=overlap,
247 mask_downsample_ratio=mask_downsample_ratio,
248 )[0]
249
250 seen = 0
251 confusion_matrix = ConfusionMatrix(nc=nc)
252 names = model.names if hasattr(model, "names") else model.module.names # get class names
253 if isinstance(names, (list, tuple)): # old format
254 names = dict(enumerate(names))
255 class_map = coco80_to_coco91_class() if is_coco else list(range(1000))
256 s = ("%22s" + "%11s" * 10) % (
257 "Class",
258 "Images",
259 "Instances",
260 "Box(P",
261 "R",
262 "mAP50",
263 "mAP50-95)",
264 "Mask(P",
265 "R",
266 "mAP50",
267 "mAP50-95)",
268 )
269 dt = Profile(device=device), Profile(device=device), Profile(device=device)
270 metrics = Metrics()
271 loss = torch.zeros(4, device=device)
272 jdict, stats = [], []
273 # callbacks.run('on_val_start')
274 pbar = tqdm(dataloader, desc=s, bar_format=TQDM_BAR_FORMAT) # progress bar
275 for batch_i, (im, targets, paths, shapes, masks) in enumerate(pbar):
276 # callbacks.run('on_val_batch_start')
277 with dt[0]:
278 if cuda:
279 im = im.to(device, non_blocking=True)
280 targets = targets.to(device)
281 masks = masks.to(device)
282 masks = masks.float()
283 im = im.half() if half else im.float() # uint8 to fp16/32
284 im /= 255 # 0 - 255 to 0.0 - 1.0
285 nb, _, height, width = im.shape # batch size, channels, height, width
286
287 # Inference
288 with dt[1]:
289 preds, protos, train_out = model(im) if compute_loss else (*model(im, augment=augment)[:2], None)
290
291 # Loss
292 if compute_loss:
293 loss += compute_loss((train_out, protos), targets, masks)[1] # box, obj, cls
294
295 # NMS
296 targets[:, 2:] *= torch.tensor((width, height, width, height), device=device) # to pixels
297 lb = [targets[targets[:, 0] == i, 1:] for i in range(nb)] if save_hybrid else [] # for autolabelling
298 with dt[2]:
299 preds = non_max_suppression(
300 preds, conf_thres, iou_thres, labels=lb, multi_label=True, agnostic=single_cls, max_det=max_det, nm=nm
301 )
302
303 # Metrics
304 plot_masks = [] # masks for plotting
305 for si, (pred, proto) in enumerate(zip(preds, protos)):
306 labels = targets[targets[:, 0] == si, 1:]
307 nl, npr = labels.shape[0], pred.shape[0] # number of labels, predictions
308 path, shape = Path(paths[si]), shapes[si][0]
309 correct_masks = torch.zeros(npr, niou, dtype=torch.bool, device=device) # init
310 correct_bboxes = torch.zeros(npr, niou, dtype=torch.bool, device=device) # init
311 seen += 1
312
313 if npr == 0:
314 if nl:
315 stats.append((correct_masks, correct_bboxes, *torch.zeros((2, 0), device=device), labels[:, 0]))
316 if plots:
317 confusion_matrix.process_batch(detections=None, labels=labels[:, 0])
318 continue
319
320 # Masks
321 midx = [si] if overlap else targets[:, 0] == si
322 gt_masks = masks[midx]
323 pred_masks = process(proto, pred[:, 6:], pred[:, :4], shape=im[si].shape[1:])
324
325 # Predictions
326 if single_cls:
327 pred[:, 5] = 0
328 predn = pred.clone()
329 scale_boxes(im[si].shape[1:], predn[:, :4], shape, shapes[si][1]) # native-space pred
330
331 # Evaluate
332 if nl:
333 tbox = xywh2xyxy(labels[:, 1:5]) # target boxes
334 scale_boxes(im[si].shape[1:], tbox, shape, shapes[si][1]) # native-space labels
335 labelsn = torch.cat((labels[:, 0:1], tbox), 1) # native-space labels
336 correct_bboxes = process_batch(predn, labelsn, iouv)
337 correct_masks = process_batch(predn, labelsn, iouv, pred_masks, gt_masks, overlap=overlap, masks=True)
338 if plots:
339 confusion_matrix.process_batch(predn, labelsn)
340 stats.append((correct_masks, correct_bboxes, pred[:, 4], pred[:, 5], labels[:, 0])) # (conf, pcls, tcls)
341
342 pred_masks = torch.as_tensor(pred_masks, dtype=torch.uint8)
343 if plots and batch_i < 3:
344 plot_masks.append(pred_masks[:15]) # filter top 15 to plot
345
346 # Save/log
347 if save_txt:
348 save_one_txt(predn, save_conf, shape, file=save_dir / "labels" / f"{path.stem}.txt")
349 if save_json:
350 pred_masks = scale_image(
351 im[si].shape[1:], pred_masks.permute(1, 2, 0).contiguous().cpu().numpy(), shape, shapes[si][1]
352 )
353 save_one_json(predn, jdict, path, class_map, pred_masks) # append to COCO-JSON dictionary
354 # callbacks.run('on_val_image_end', pred, predn, path, names, im[si])
355
356 # Plot images
357 if plots and batch_i < 3:
358 if len(plot_masks):
359 plot_masks = torch.cat(plot_masks, dim=0)
360 plot_images_and_masks(im, targets, masks, paths, save_dir / f"val_batch{batch_i}_labels.jpg", names)
361 plot_images_and_masks(
362 im,
363 output_to_target(preds, max_det=15),
364 plot_masks,
365 paths,
366 save_dir / f"val_batch{batch_i}_pred.jpg",
367 names,
368 ) # pred
369
370 # callbacks.run('on_val_batch_end')
371
372 # Compute metrics
373 stats = [torch.cat(x, 0).cpu().numpy() for x in zip(*stats)] # to numpy
374 if len(stats) and stats[0].any():
375 results = ap_per_class_box_and_mask(*stats, plot=plots, save_dir=save_dir, names=names)
376 metrics.update(results)
377 nt = np.bincount(stats[4].astype(int), minlength=nc) # number of targets per class
378
379 # Print results
380 pf = "%22s" + "%11i" * 2 + "%11.3g" * 8 # print format
381 LOGGER.info(pf % ("all", seen, nt.sum(), *metrics.mean_results()))
382 if nt.sum() == 0:
383 LOGGER.warning(f"WARNING ⚠️ no labels found in {task} set, can not compute metrics without labels")
384
385 # Print results per class
386 if (verbose or (nc < 50 and not training)) and nc > 1 and len(stats):
387 for i, c in enumerate(metrics.ap_class_index):
388 LOGGER.info(pf % (names[c], seen, nt[c], *metrics.class_result(i)))
389
390 # Print speeds
391 t = tuple(x.t / seen * 1e3 for x in dt) # speeds per image
392 if not training:
393 shape = (batch_size, 3, imgsz, imgsz)
394 LOGGER.info(f"Speed: %.1fms pre-process, %.1fms inference, %.1fms NMS per image at shape {shape}" % t)
395
396 # Plots
397 if plots:
398 confusion_matrix.plot(save_dir=save_dir, names=list(names.values()))
399 # callbacks.run('on_val_end')
400
401 mp_bbox, mr_bbox, map50_bbox, map_bbox, mp_mask, mr_mask, map50_mask, map_mask = metrics.mean_results()
402
403 # Save JSON
404 if save_json and len(jdict):
405 w = Path(weights[0] if isinstance(weights, list) else weights).stem if weights is not None else "" # weights
406 anno_json = str(Path("../datasets/coco/annotations/instances_val2017.json")) # annotations
407 pred_json = str(save_dir / f"{w}_predictions.json") # predictions
408 LOGGER.info(f"\nEvaluating pycocotools mAP... saving {pred_json}...")
409 with open(pred_json, "w") as f:
410 json.dump(jdict, f)
411
412 try: # https://github.com/cocodataset/cocoapi/blob/master/PythonAPI/pycocoEvalDemo.ipynb

414 from pycocotools.cocoeval import COCOeval
415
416 anno = COCO(anno_json) # init annotations api
417 pred = anno.loadRes(pred_json) # init predictions api
418 results = []
419 for eval in COCOeval(anno, pred, "bbox"), COCOeval(anno, pred, "segm"):

421 eval.params.imgIds = [int(Path(x).stem) for x in dataloader.dataset.im_files] # img ID to evaluate

430 # Return results
431 model.float() # for training
432 if not training:
433 s = f"\n{len(list(save_dir.glob('labels/*.txt')))} labels saved to {save_dir / 'labels'}" if save_txt else ""
434 LOGGER.info(f"Results saved to {colorstr('bold', save_dir)}{s}")
435 final_metric = mp_bbox, mr_bbox, map50_bbox, map_bbox, mp_mask, mr_mask, map50_mask, map_mask
436 return (*final_metric, *(loss.cpu() / len(dataloader)).tolist()), metrics.get_maps(nc), t

471def main(opt):
472 check_requirements(ROOT / "requirements.txt", exclude=("tensorboard", "thop"))
473
474 if opt.task in ("train", "val", "test"): # run normally
475 if opt.conf_thres > 0.001: # https://github.com/ultralytics/yolov5/issues/1466
476 LOGGER.warning(f"WARNING ⚠️ confidence threshold {opt.conf_thres} > 0.001 produces invalid results")
477 if opt.save_hybrid:
478 LOGGER.warning("WARNING ⚠️ --save-hybrid returns high mAP from hybrid labels, not from predictions alone")
479 run(**vars(opt))
480
481 else:
482 weights = opt.weights if isinstance(opt.weights, list) else [opt.weights]
483 opt.half = torch.cuda.is_available() and opt.device != "cpu" # FP16 for fastest results
484 if opt.task == "speed": # speed benchmarks
485 # python val.py --task speed --data coco.yaml --batch 1 --weights yolov5n.pt yolov5s.pt...
486 opt.conf_thres, opt.iou_thres, opt.save_json = 0.25, 0.45, False
487 for opt.weights in weights:
488 run(**vars(opt), plots=False)
489
490 elif opt.task == "study": # speed vs mAP benchmarks
491 # python val.py --task study --data coco.yaml --iou 0.7 --weights yolov5n.pt yolov5s.pt...
492 for opt.weights in weights:
493 f = f"study_{Path(opt.data).stem}_{Path(opt.weights).stem}.txt" # filename to save to
494 x, y = list(range(256, 1536 + 128, 128)), [] # x axis (image sizes), y axis
495 for opt.imgsz in x: # img-size
496 LOGGER.info(f"\nRunning {f} --imgsz {opt.imgsz}...")
