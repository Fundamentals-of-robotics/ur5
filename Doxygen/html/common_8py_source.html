60def autopad(k, p=None, d=1): # kernel, padding, dilation
61 # Pad to 'same' shape outputs
62 if d > 1:
63 k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k] # actual kernel-size
64 if p is None:
65 p = k // 2 if isinstance(k, int) else [x // 2 for x in k] # auto-pad
66 return p

69class Conv(nn.Module):
70 # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
71 default_act = nn.SiLU() # default activation
72
73 def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
74 super().__init__()
75 self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
76 self.bn = nn.BatchNorm2d(c2)
77 self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
78
79 def forward(self, x):
80 return self.act(self.bn(self.conv(x)))
81
82 def forward_fuse(self, x):
83 return self.act(self.conv(x))

98class TransformerLayer(nn.Module):
99 # Transformer layer https://arxiv.org/abs/2010.11929 (LayerNorm layers removed for better performance)
100 def __init__(self, c, num_heads):
101 super().__init__()
102 self.q = nn.Linear(c, c, bias=False)
103 self.k = nn.Linear(c, c, bias=False)
104 self.v = nn.Linear(c, c, bias=False)
105 self.ma = nn.MultiheadAttention(embed_dim=c, num_heads=num_heads)
106 self.fc1 = nn.Linear(c, c, bias=False)
107 self.fc2 = nn.Linear(c, c, bias=False)
108
109 def forward(self, x):
110 x = self.ma(self.q(x), self.k(x), self.v(x))[0] + x
111 x = self.fc2(self.fc1(x)) + x
112 return x

115class TransformerBlock(nn.Module):
116 # Vision Transformer https://arxiv.org/abs/2010.11929
117 def __init__(self, c1, c2, num_heads, num_layers):
118 super().__init__()
119 self.conv = None
120 if c1 != c2:
121 self.conv = Conv(c1, c2)
122 self.linear = nn.Linear(c2, c2) # learnable position embedding
123 self.tr = nn.Sequential(*(TransformerLayer(c2, num_heads) for _ in range(num_layers)))
124 self.c2 = c2
125
126 def forward(self, x):
127 if self.conv is not None:
128 x = self.conv(x)
129 b, _, w, h = x.shape
130 p = x.flatten(2).permute(2, 0, 1)
131 return self.tr(p + self.linear(p)).permute(1, 2, 0).reshape(b, self.c2, w, h)

134class Bottleneck(nn.Module):
135 # Standard bottleneck
136 def __init__(self, c1, c2, shortcut=True, g=1, e=0.5): # ch_in, ch_out, shortcut, groups, expansion
137 super().__init__()
138 c_ = int(c2 * e) # hidden channels
139 self.cv1 = Conv(c1, c_, 1, 1)
140 self.cv2 = Conv(c_, c2, 3, 1, g=g)
141 self.add = shortcut and c1 == c2
142
143 def forward(self, x):
144 return x + self.cv2(self.cv1(x)) if self.add else self.cv2(self.cv1(x))

147class BottleneckCSP(nn.Module):
148 # CSP Bottleneck https://github.com/WongKinYiu/CrossStagePartialNetworks
149 def __init__(self, c1, c2, n=1, shortcut=True, g=1, e=0.5): # ch_in, ch_out, number, shortcut, groups, expansion
150 super().__init__()
151 c_ = int(c2 * e) # hidden channels
152 self.cv1 = Conv(c1, c_, 1, 1)
153 self.cv2 = nn.Conv2d(c1, c_, 1, 1, bias=False)
154 self.cv3 = nn.Conv2d(c_, c_, 1, 1, bias=False)
155 self.cv4 = Conv(2 * c_, c2, 1, 1)
156 self.bn = nn.BatchNorm2d(2 * c_) # applied to cat(cv2, cv3)
157 self.act = nn.SiLU()
158 self.m = nn.Sequential(*(Bottleneck(c_, c_, shortcut, g, e=1.0) for _ in range(n)))
159
160 def forward(self, x):
161 y1 = self.cv3(self.m(self.cv1(x)))
162 y2 = self.cv2(x)
163 return self.cv4(self.act(self.bn(torch.cat((y1, y2), 1))))

180class C3(nn.Module):
181 # CSP Bottleneck with 3 convolutions
182 def __init__(self, c1, c2, n=1, shortcut=True, g=1, e=0.5): # ch_in, ch_out, number, shortcut, groups, expansion
183 super().__init__()
184 c_ = int(c2 * e) # hidden channels
185 self.cv1 = Conv(c1, c_, 1, 1)
186 self.cv2 = Conv(c1, c_, 1, 1)

190 def forward(self, x):
191 return self.cv3(torch.cat((self.m(self.cv1(x)), self.cv2(x)), 1))

226class SPP(nn.Module):
227 # Spatial Pyramid Pooling (SPP) layer https://arxiv.org/abs/1406.4729
228 def __init__(self, c1, c2, k=(5, 9, 13)):
229 super().__init__()

231 self.cv1 = Conv(c1, c_, 1, 1)

238 warnings.simplefilter("ignore") # suppress torch 1.9.0 max_pool2d() warning
239 return self.cv2(torch.cat([x] + [m(x) for m in self.m], 1))

243 # Spatial Pyramid Pooling - Fast (SPPF) layer for YOLOv5 by Glenn Jocher

285class GhostBottleneck(nn.Module):
286 # Ghost Bottleneck https://github.com/huawei-noah/ghostnet
287 def __init__(self, c1, c2, k=3, s=1): # ch_in, ch_out, kernel, stride
288 super().__init__()
289 c_ = c2 // 2
290 self.conv = nn.Sequential(
291 GhostConv(c1, c_, 1, 1), # pw
292 DWConv(c_, c_, k, s, act=False) if s == 2 else nn.Identity(), # dw
293 GhostConv(c_, c2, 1, 1, act=False),
294 ) # pw-linear
295 self.shortcut = (
296 nn.Sequential(DWConv(c1, c1, k, s, act=False), Conv(c1, c2, 1, 1, act=False)) if s == 2 else nn.Identity()
297 )
298
299 def forward(self, x):
300 return self.conv(x) + self.shortcut(x)

303class Contract(nn.Module):
304 # Contract width-height into channels, i.e. x(1,64,80,80) to x(1,256,40,40)
305 def __init__(self, gain=2):
306 super().__init__()
307 self.gain = gain
308
309 def forward(self, x):
310 b, c, h, w = x.size() # assert (h / s == 0) and (W / s == 0), 'Indivisible gain'
311 s = self.gain
312 x = x.view(b, c, h // s, s, w // s, s) # x(1,64,40,2,40,2)
313 x = x.permute(0, 3, 5, 1, 2, 4).contiguous() # x(1,2,2,64,40,40)
314 return x.view(b, c * s * s, h // s, w // s) # x(1,256,40,40)

317class Expand(nn.Module):
318 # Expand channels into width-height, i.e. x(1,64,80,80) to x(1,16,160,160)
319 def __init__(self, gain=2):
320 super().__init__()
321 self.gain = gain
322
323 def forward(self, x):
324 b, c, h, w = x.size() # assert C / s ** 2 == 0, 'Indivisible gain'
325 s = self.gain
326 x = x.view(b, s, s, c // s**2, h, w) # x(1,2,2,16,80,80)
327 x = x.permute(0, 3, 4, 1, 5, 2).contiguous() # x(1,16,80,2,80,2)
328 return x.view(b, c // s**2, h * s, w * s) # x(1,16,160,160)

694 @smart_inference_mode()
695 def forward(self, ims, size=640, augment=False, profile=False):
696 # Inference from various sources. For size(height=640, width=1280), RGB images example inputs are:
697 # file: ims = 'data/images/zidane.jpg' # str or PosixPath
698 # URI: = 'https://ultralytics.com/images/zidane.jpg'
699 # OpenCV: = cv2.imread('image.jpg')[:,:,::-1] # HWC BGR to RGB x(640,1280,3)
700 # PIL: = Image.open('image.jpg') or ImageGrab.grab() # HWC x(640,1280,3)
701 # numpy: = np.zeros((640,1280,3)) # HWC
702 # torch: = torch.zeros(16,3,320,640) # BCHW (scaled to size=640, 0-1 values)
703 # multiple: = [Image.open('image1.jpg'), Image.open('image2.jpg'), ...] # list of images
704
705 dt = (Profile(), Profile(), Profile())
706 with dt[0]:
707 if isinstance(size, int): # expand
708 size = (size, size)
709 p = next(self.model.parameters()) if self.pt else torch.empty(1, device=self.model.device) # param
710 autocast = self.amp and (p.device.type != "cpu") # Automatic Mixed Precision (AMP) inference
711 if isinstance(ims, torch.Tensor): # torch
712 with amp.autocast(autocast):
713 return self.model(ims.to(p.device).type_as(p), augment=augment) # inference
714
715 # Pre-process
716 n, ims = (len(ims), list(ims)) if isinstance(ims, (list, tuple)) else (1, [ims]) # number, list of images
717 shape0, shape1, files = [], [], [] # image and inference shapes, filenames
718 for i, im in enumerate(ims):
719 f = f"image{i}" # filename
720 if isinstance(im, (str, Path)): # filename or uri
721 im, f = Image.open(requests.get(im, stream=True).raw if str(im).startswith("http") else im), im
722 im = np.asarray(exif_transpose(im))
723 elif isinstance(im, Image.Image): # PIL Image
724 im, f = np.asarray(exif_transpose(im)), getattr(im, "filename", f) or f
725 files.append(Path(f).with_suffix(".jpg").name)
726 if im.shape[0] < 5: # image in CHW
727 im = im.transpose((1, 2, 0)) # reverse dataloader .transpose(2, 0, 1)
728 im = im[..., :3] if im.ndim == 3 else cv2.cvtColor(im, cv2.COLOR_GRAY2BGR) # enforce 3ch input
729 s = im.shape[:2] # HWC
730 shape0.append(s) # image shape
731 g = max(size) / max(s) # gain
732 shape1.append([int(y * g) for y in s])
733 ims[i] = im if im.data.contiguous else np.ascontiguousarray(im) # update
734 shape1 = [make_divisible(x, self.stride) for x in np.array(shape1).max(0)] # inf shape
735 x = [letterbox(im, shape1, auto=False)[0] for im in ims] # pad
736 x = np.ascontiguousarray(np.array(x).transpose((0, 3, 1, 2))) # stack and BHWC to BCHW
737 x = torch.from_numpy(x).to(p.device).type_as(p) / 255 # uint8 to fp16/32
738
739 with amp.autocast(autocast):
740 # Inference
741 with dt[1]:
742 y = self.model(x, augment=augment) # forward
743
744 # Post-process
745 with dt[2]:
746 y = non_max_suppression(
747 y if self.dmb else y[0],
748 self.confconf,
749 self.iouiou,
750 self.classes,
751 self.agnosticagnostic,
752 self.multi_labelmulti_label,
753 max_det=self.max_det,
754 ) # NMS
755 for i in range(n):
756 scale_boxes(shape1, y[i][:, :4], shape0[i])
757
758 return Detections(ims, y, files, dt, self.names, x.shape)
